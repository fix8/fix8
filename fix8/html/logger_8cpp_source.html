71 int Logger::operator()()
72 {
73  unsigned received(0);
74  bool stopping(false);
75 
76  for (;;)
77  {
78  LogElement msg;
79  if (stopping) // make sure we dequeue any pending msgs before exiting
80  {
81  if (!_msg_queue.try_pop(msg))
82  break;
83  }
84  else
85  _msg_queue.pop (msg); // will block
86 
87  ++received;
88 
89  if (msg._str.empty()) // means exit
90  {
91  stopping = true;
92  continue;
93  }
94 
95  ostringstream ostr;
96 
97  if (_flags & sequence)
98  {
99  ostr << setw(7) << right << setfill('0');
100  if (_flags & direction)
101  ostr << (msg._val ? ++_sequence : ++_osequence) << ' ';
102  else
103  ostr << ++_sequence << ' ';
104  }
105 
106  if (_flags & thread)
107  ostr << get_thread_code(msg._tid) << ' ';
108 
109  if (_flags & direction)
110  ostr << (msg._val ? " in" : "out") << ' ' ;
111 
112  if (_flags & timestamp)
113  {
114  string ts;
115  ostr << GetTimeAsStringMS(ts, &msg._when, 9) << ' ';
116  }
117 
118  if (_flags & buffer)
119  {
120  string result(ostr.str());
121  result += msg._str;
122  _buffer.push_back(result);
123  }
124  else
125  {
126  tbb::mutex::scoped_lock guard(_mutex);
127  get_stream() << ostr.str() << msg._str << endl;
128  }
129  }
130 
131  return 0;
132 }

145 void Logger::purge_thread_codes()
146 {
147  tbb::mutex::scoped_lock guard(_mutex);
148 
149  for (ThreadCodes::iterator itr(_thread_codes.begin()); itr != _thread_codes.end();)
150  {
151  // a little trick to see if a thread is still alive
152  clockid_t clock_id;
153  if (pthread_getcpuclockid(itr->first, &clock_id) == ESRCH)
154  {
155  _rev_thread_codes.erase(itr->second);
156  _thread_codes.erase(itr++); // post inc, takes copy before incr;
157  }
158  else
159  ++itr;
160  }
161 }

164 char Logger::get_thread_code(pthread_t tid)
165 {
166  tbb::mutex::scoped_lock guard(_mutex);
167 
168  ThreadCodes::const_iterator itr(_thread_codes.find(tid));
169  if (itr != _thread_codes.end())
170  return itr->second;
171 
172  for (char acode('A'); acode < 127; ++acode) // A-~ will allow for 86 threads
173  {
174  RevThreadCodes::const_iterator itr(_rev_thread_codes.find(acode));
175  if (itr == _rev_thread_codes.end())
176  {
177  _thread_codes.insert(ThreadCodes::value_type(tid, acode));
178  _rev_thread_codes.insert(ThreadCodes::value_type(acode, tid));
179  return acode;
180  }
181  }
182 
183  return '?';
184 }

198 bool FileLogger::rotate(bool force)
199 {
200  tbb::mutex::scoped_lock guard(_mutex);
201 
202  string thislFile(_pathname);
203 #ifdef HAVE_COMPRESSION
204  if (_flags & compress)
205  thislFile += ".gz";
206 #endif
207  if (_rotnum > 0 && (!_flags.has(append) || force))
208  {
209  vector<string> rlst(_rotnum);
210  rlst.push_back(thislFile);
211 
212  for (unsigned ii(0); ii < _rotnum && ii < max_rotation; ++ii)
213  {
214  ostringstream ostr;
215  ostr << _pathname << '.' << (ii + 1);
216  if (_flags & compress)
217  ostr << ".gz";
218  rlst.push_back(ostr.str());
219  }
220 
221  for (unsigned ii(_rotnum); ii; --ii)
222  rename (rlst[ii - 1].c_str(), rlst[ii].c_str()); // ignore errors
223  }
224 
225  delete _ofs;
226 
227  const ios_base::openmode mode (_flags & append ? ios_base::out | ios_base::app : ios_base::out);
228 #ifdef HAVE_COMPRESSION
229  if (_flags & compress)
230  _ofs = new ogzstream(thislFile.c_str(), mode);
231  else
232 #endif
233  _ofs = new ofstream(thislFile.c_str(), mode);
234 
235  return true;
236 }

239 PipeLogger::PipeLogger(const string& fname, const ebitset<Flags> flags) : Logger(flags)
240 {
241  const string pathname(fname.substr(1));
242 
243  // | uses IFS safe cfpopen; ! uses old popen if available
244  FILE *pcmd(
245 #ifdef HAVE_POPEN
246  fname[0] == '!' ? popen(pathname.c_str(), "w") :
247 #endif

249  if (pcmd == 0)
250 #ifdef DEBUG

252 #endif
253  ;
254  else if (ferror(pcmd))
255 #ifdef DEBUG
256  *errofs << pathname << " shows ferror" << endl
257 #endif
258  ;
259  else
260  {
261  _ofs = new fptrostream(pcmd, fname[0] == '|');
262  _flags |= pipe;
263  }
