60 {

63 
64  for (; !_session.is_shutdown();)

68  f8String msg;
69

94  }
95  catch (exception& e) // also catches Poco::Net::NetException

97  _session.log(e.what());
98  ++invalid;
99  }
100  }
101 
102  ostringstream ostr;

104  << invalid << " invalid";

150 bool FIXReader::read(f8String& to) // read a complete FIX message
151 {
152  char msg_buf[_max_msg_len] = {};
153  int result(sockRead(msg_buf, _bg_sz));
154 
155  if (result == static_cast<int>(_bg_sz))
156  {
157  char bt;
158  size_t offs(_bg_sz);
159  do // get the last chrs of bodylength and ^A
160  {
161  if (sockRead(&bt, 1) != 1)
162  return false;
163  if (!isdigit(bt) && bt != default_field_separator)
164  throw IllegalMessage(msg_buf);
165  msg_buf[offs++] = bt;
166  }
167  while (bt != default_field_separator && offs < _max_msg_len);
168  to.assign(msg_buf, offs);
169 
170  f8String tag, bgstr, len;
171  unsigned result;
172  if ((result = MessageBase::extract_element(to.data(), to.size(), tag, bgstr)))
173  {
174  if (tag != "8")
175  throw IllegalMessage(to);
176 
177  if (bgstr != _session.get_ctx()._beginStr) // invalid FIX version
178  throw InvalidVersion(bgstr);
179 
180  if ((result = MessageBase::extract_element(to.data() + result, to.size() - result, tag, len)))
181  {
182  if (tag != "9")
183  throw IllegalMessage(to);
184 
185  const unsigned mlen(fast_atoi<unsigned>(len.c_str()));
186  if (mlen == 0 || mlen > _max_msg_len - _bg_sz - _chksum_sz) // invalid msglen
187  throw InvalidBodyLength(mlen);
188 
189  // read the body
190  if ((result = sockRead(msg_buf, mlen) != static_cast<int>(mlen)))
191  return false;
192 
193  // read the checksum
194  if ((result = sockRead(msg_buf + mlen, _chksum_sz) != static_cast<int>(_chksum_sz)))
195  return false;
196 
197  to.append(msg_buf, mlen + _chksum_sz);
198  _session.update_received();
199  //string ts;
200  //cerr << GetTimeAsStringMS(ts, &_session.get_last_received(), 9) << endl;
201  return true;
202  }
203  }
204 
205  throw IllegalMessage(to);
206  }
207 
208  return false;
209 }

214 {
215  int result(0), processed(0), invalid(0);
216 
217  for (; !_session.is_shutdown();)
218  {
219  try
220  {
221  Message *inmsg(0);

225 #if defined MSGRECYCLING

276 bool ClientConnection::connect()
277 {
278  unsigned attempts(0);
279  const LoginParameters& lparam(_session.get_login_parameters());
280  Poco::Timespan timeout(1000000);
281 
282  while (attempts < lparam._login_retries)
283  {
284  ostringstream ostr;
285 
286  try
287  {
288  ostr.str("");
289  ostr << "Trying to connect to: " << _addr.toString() << " (" << ++attempts << ')';
290  _session.log(ostr.str());
291  const LoginParameters& lparam(_session.get_login_parameters());
292  _sock->connect(_addr, timeout);
293  if (lparam._recv_buf_sz)
294  set_recv_buf_sz(lparam._recv_buf_sz);
295  if (lparam._send_buf_sz)
296  set_send_buf_sz(lparam._send_buf_sz);
297  _sock->setLinger(false, 0);
298  _sock->setNoDelay(_no_delay);
299  _session.log("Connection successful");
300  return _connected = true;
301  }
302  catch (exception& e) // also catches Poco::Net::NetException
303  {
304  ostr.str("");
305  ostr << "exception: ";
306  if (dynamic_cast<Poco::Exception*>(&e))
307  ostr << (static_cast<Poco::Exception&>(e)).displayText();
308  else
309  ostr << e.what();
310  _session.log(ostr.str());
311  millisleep(lparam._login_retry_interval);
312  }
313  }
314 
315  _session.log("Connection failed");
316  return false;
317 }
