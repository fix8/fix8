92  unsigned load_map(const std::string& tag, ConfigMap& map_name, const bool is_session=false)
93  {
94  XmlElement::XmlSet slist;
95  if (_root->find(tag, slist))
96  {
97  for(XmlElement::XmlSet::const_iterator itr(slist.begin()); itr != slist.end(); ++itr)
98  {
99  std::string name;
100  if ((*itr)->GetAttr("name", name) && is_session ? (*itr)->FindAttr("active", false) : true)
101  {
102  map_name.insert(ConfigMap::value_type(name, *itr));
103  if (is_session)
104  _allsessions.push_back(*itr);
105  }
106  }
107  }
108 
109  return map_name.size();
110  }
