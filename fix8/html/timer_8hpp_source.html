129 template<typename T>
130 int Timer<T>::operator()()
131 {
132  unsigned elapsed(0);
133 
134  while(true)
135  {
136  bool shouldsleep(false);
137  {
138  f8_scoped_lock guard(_mutex);
139 
140  if (_event_queue.size())
141  {
142  const TimerEvent<T>& op(_event_queue.top());
143  if (!op._t) // empty timeval means exit
144  {
145  _event_queue.pop(); // remove from queue
146  break;
147  }
148 
149  if (op._t <= Tickval::get_tickval()) // has elapsed
150  {
151  const TimerEvent<T> rop(_event_queue.top()); // take a copy
152  _event_queue.pop(); // remove from queue
153  guard.release();
154  ++elapsed;
155  if (!(_monitor.*rop._callback)())
156  break;
157  }
158  else
159  shouldsleep = true;
160  }
161  else
162  shouldsleep = true;
163  } // we want the lock to go out of scope before we sleep
164 
165  if (shouldsleep)
166  hypersleep<h_milliseconds>(_granularity);
167  }
168 
169  std::ostringstream ostr;
170  ostr << "Terminating Timer thread (" << elapsed << " elapsed, " << _event_queue.size() << " queued).";
171  GlobalLogger::instance()->send(ostr.str());
172  return 0;
173 }

192 template<typename T>
193 bool Timer<T>::schedule(const TimerEvent<T>& what, const unsigned timeToWait)
194 {
195  Tickval tofire;
196 
197  if (timeToWait)
198  {
199  // Calculate time to fire
200  Tickval::get_tickval(tofire);
201  tofire += timeToWait * Tickval::million;
202  }
203 
204  what.set(tofire);
205  f8_scoped_lock guard(_mutex);
206  _event_queue.push(what);
207 
208  return true;
209 }
