273 static inline int ff_mapThreadToCpu(int cpu_id, int priority_level=0) {
274  if (cpu_id > ff_numCores()) return EINVAL;
275 #if defined(__linux__) && defined(CPU_SET)
276  cpu_set_t mask;
277  CPU_ZERO(&mask);
278  CPU_SET(cpu_id, &mask);
279  if (sched_setaffinity(gettid(), sizeof(mask), &mask) != 0)
280  return EINVAL;
281  return (ff_setPriority(priority_level));
282 #elif defined(__APPLE__) && MAC_OS_X_HAS_AFFINITY
283  // Mac OS does not implement direct pinning of threads onto cores.
284  // Threads can be organised in affinity set. Using requested CPU
285  // tag for the set. Cores under the same L2 cache are not distinguished.
286  // Should be called before running the thread.
287 #define CACHE_LEVELS 3
288  #define CACHE_L2 2
289  size_t len;
290 
291  if (sysctlbyname("hw.cacheconfig",NULL, &len, NULL, 0) != 0) {
292  perror("sysctl");
293  } else {
294  int64_t cacheconfig[len];
295  if (sysctlbyname("hw.cacheconfig", &cacheconfig[0], &len, NULL, 0) != 0)
296  perror("sysctl: unable to get hw.cacheconfig");
297  else {
298  /*
299  for (size_t i=0;i<CACHE_LEVELS;i++)
300  std::cerr << " Cache " << i << " shared by " << cacheconfig[i] << " cores\n";
301  */
302  struct thread_affinity_policy mypolicy;
303  // Define sets taking in account pinning is performed on L2
304  mypolicy.affinity_tag = cpu_id/cacheconfig[CACHE_L2];
305  if ( thread_policy_set(mach_thread_self(), THREAD_AFFINITY_POLICY, (integer_t*) &mypolicy, THREAD_AFFINITY_POLICY_COUNT) != KERN_SUCCESS ) {
306  std::cerr << "Setting affinity of thread ? (" << mach_thread_self() << ") failed!" << std::endl;
307  return EINVAL;
308  } // else {
309  // std::cerr << "Sucessfully set affinity of thread (" <<
310  // mach_thread_self() << ") to core " << cpu_id/cacheconfig[CACHE_L2] << "\n";
311  // }
312  }
313  }
314 
315  return(ff_setPriority(priority_level));
316 #elif (defined(_MSC_VER) || defined(__INTEL_COMPILER)) && defined(_WIN32)
317  if (-1==SetThreadIdealProcessor(GetCurrentThread(),cpu_id)) {
318  perror("ff_mapThreadToCpu:SetThreadIdealProcessor");
319  return EINVAL;
320  }
321  //std::cerr << "Successfully set affinity of thread " << GetCurrentThreadId() << " to core " << cpu_id << "\n";
322 #else
323 #warning "CPU_SET not defined, cannot map thread to specific CPU"
324 #endif
325  return 0;
326 }
