489 inline size_t itoa(int value, char *result, const int base=10)
490 {
491  // check that the base if valid
492  if (base < 2 || base > 36)
493  {
494  *result = 0;
495  return 0;
496  }
497 
498  char *ptr(result), *ptr1(result);
499  int tmp_value;
500 
501  do
502  {
503  tmp_value = value;
504  value /= base;
505  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
506  }
507  while (value);
508 
509  // Apply negative sign
510  if (tmp_value < 0)
511  *ptr++ = '-';
512  *ptr-- = 0;
513  while(ptr1 < ptr)
514  {
515  const char tmp_char(*ptr);
516  *ptr-- = *ptr1;
517  *ptr1++ = tmp_char;
518  }
519  return ::strlen(result);
520 }

532 inline double fast_atof (const char *p)
533 {
534  bool frac(false);
535  double sign(1.), value(0.), scale(1.);
536 
537  while (isspace(*p))
538  ++p;
539 
540  // Get sign, if any.
541  if (*p == '-')
542  {
543  sign = -1.;
544  ++p;
545  }
546  else if (*p == '+')
547  ++p;
548 
549  // Get digits before decimal point or exponent, if any.
550  while (isdigit(*p))
551  {
552  value = value * 10. + (*p - '0');
553  ++p;
554  }
555 
556  // Get digits after decimal point, if any.
557  if (*p == '.')
558  {
559  ++p;
560  double pow10(10.);
561  while (isdigit(*p))
562  {
563  value += (*p - '0') / pow10;
564  pow10 *= 10.;
565  ++p;
566  }
567  }
568 
569  // Handle exponent, if any.
570  if (toupper(*p) == 'E')
571  {
572  unsigned int expon(0);
573  ++p;
574 
575  // Get sign of exponent, if any.
576  if (*p == '-')
577  {
578  frac = true;
579  ++p;
580  }
581  else if (*p == '+')
582  ++p;
583 
584  // Get digits of exponent, if any.
585  while (isdigit(*p))
586  {
587  expon = expon * 10 + (*p - '0');
588  ++p;
589  }
590  if (expon > 308)
591  expon = 308;
592 
593  // Calculate scaling factor.
594  while (expon >= 50)
595  {
596  scale *= 1E50;
597  expon -= 50;
598  }
599  while (expon >= 8)
600  {
601  scale *= 1E8;
602  expon -= 8;
603  }
604  while (expon > 0)
605  {
606  scale *= 10.0;
607  expon -= 1;
608  }
609  }
610 
611  // Return signed and scaled floating point result.
612  return sign * (frac ? (value / scale) : (value * scale));
613 }
