479 inline size_t itoa(int value, char *result, const int base=10)
480 {
481  // check that the base if valid
482  if (base < 2 || base > 36)
483  {
484  *result = 0;
485  return 0;
486  }
487 
488  char *ptr(result), *ptr1(result);
489  int tmp_value;
490 
491  do
492  {
493  tmp_value = value;
494  value /= base;
495  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
496  }
497  while (value);
498 
499  // Apply negative sign
500  if (tmp_value < 0)
501  *ptr++ = '-';
502  *ptr-- = 0;
503  while(ptr1 < ptr)
504  {
505  const char tmp_char(*ptr);
506  *ptr-- = *ptr1;
507  *ptr1++ = tmp_char;
508  }
509  return ::strlen(result);
510 }

522 inline double fast_atof (const char *p)
523 {
524  bool frac(false);
525  double sign(1.), value(0.), scale(1.);
526 
527  while (isspace(*p))
528  ++p;
529 
530  // Get sign, if any.
531  if (*p == '-')
532  {
533  sign = -1.;
534  ++p;
535  }
536  else if (*p == '+')
537  ++p;
538 
539  // Get digits before decimal point or exponent, if any.
540  while (isdigit(*p))
541  {
542  value = value * 10. + (*p - '0');
543  ++p;
544  }
545 
546  // Get digits after decimal point, if any.
547  if (*p == '.')
548  {
549  ++p;
550  double pow10(10.);
551  while (isdigit(*p))
552  {
553  value += (*p - '0') / pow10;
554  pow10 *= 10.;
555  ++p;
556  }
557  }
558 
559  // Handle exponent, if any.
560  if (toupper(*p) == 'E')
561  {
562  unsigned int expon(0);
563  ++p;
564 
565  // Get sign of exponent, if any.
566  if (*p == '-')
567  {
568  frac = true;
569  ++p;
570  }
571  else if (*p == '+')
572  ++p;
573 
574  // Get digits of exponent, if any.
575  while (isdigit(*p))
576  {
577  expon = expon * 10 + (*p - '0');
578  ++p;
579  }
580  if (expon > 308)
581  expon = 308;
582 
583  // Calculate scaling factor.
584  while (expon >= 50)
585  {
586  scale *= 1E50;
587  expon -= 50;
588  }
589  while (expon >= 8)
590  {
591  scale *= 1E8;
592  expon -= 8;
593  }
594  while (expon > 0)
595  {
596  scale *= 10.0;
597  expon -= 1;
598  }
599  }
600 
601  // Return signed and scaled floating point result.
602  return sign * (frac ? (value / scale) : (value * scale));
603 }
