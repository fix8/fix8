487 inline size_t itoa(int value, char *result, const int base=10)
488 {
489  // check that the base if valid
490  if (base < 2 || base > 36)
491  {
492  *result = 0;
493  return 0;
494  }
495 
496  char *ptr(result), *ptr1(result);
497  int tmp_value;
498 
499  do
500  {
501  tmp_value = value;
502  value /= base;
503  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
504  }
505  while (value);
506 
507  // Apply negative sign
508  if (tmp_value < 0)
509  *ptr++ = '-';
510  *ptr-- = 0;
511  while(ptr1 < ptr)
512  {
513  const char tmp_char(*ptr);
514  *ptr-- = *ptr1;
515  *ptr1++ = tmp_char;
516  }
517  return ::strlen(result);
518 }

530 inline double fast_atof (const char *p)
531 {
532  bool frac(false);
533  double sign(1.), value(0.), scale(1.);
534 
535  while (isspace(*p))
536  ++p;
537 
538  // Get sign, if any.
539  if (*p == '-')
540  {
541  sign = -1.;
542  ++p;
543  }
544  else if (*p == '+')
545  ++p;
546 
547  // Get digits before decimal point or exponent, if any.
548  while (isdigit(*p))
549  {
550  value = value * 10. + (*p - '0');
551  ++p;
552  }
553 
554  // Get digits after decimal point, if any.
555  if (*p == '.')
556  {
557  ++p;
558  double pow10(10.);
559  while (isdigit(*p))
560  {
561  value += (*p - '0') / pow10;
562  pow10 *= 10.;
563  ++p;
564  }
565  }
566 
567  // Handle exponent, if any.
568  if (toupper(*p) == 'E')
569  {
570  unsigned int expon(0);
571  ++p;
572 
573  // Get sign of exponent, if any.
574  if (*p == '-')
575  {
576  frac = true;
577  ++p;
578  }
579  else if (*p == '+')
580  ++p;
581 
582  // Get digits of exponent, if any.
583  while (isdigit(*p))
584  {
585  expon = expon * 10 + (*p - '0');
586  ++p;
587  }
588  if (expon > 308)
589  expon = 308;
590 
591  // Calculate scaling factor.
592  while (expon >= 50)
593  {
594  scale *= 1E50;
595  expon -= 50;
596  }
597  while (expon >= 8)
598  {
599  scale *= 1E8;
600  expon -= 8;
601  }
602  while (expon > 0)
603  {
604  scale *= 10.0;
605  expon -= 1;
606  }
607  }
608 
609  // Return signed and scaled floating point result.
610  return sign * (frac ? (value / scale) : (value * scale));
611 }
