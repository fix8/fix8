532 inline size_t itoa(int value, char *result, const int base=10)
533 {
534  // check that the base if valid
535  if (base < 2 || base > 36)
536  {
537  *result = 0;
538  return 0;
539  }
540 
541  char *ptr(result), *ptr1(result);
542  int tmp_value;
543 
544  do
545  {
546  tmp_value = value;
547  value /= base;
548  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
549  }
550  while (value);
551 
552  // Apply negative sign
553  if (tmp_value < 0)
554  *ptr++ = '-';
555  *ptr-- = 0;
556  while(ptr1 < ptr)
557  {
558  const char tmp_char(*ptr);
559  *ptr-- = *ptr1;
560  *ptr1++ = tmp_char;
561  }
562  return ::strlen(result);
563 }

575 inline double fast_atof (const char *p)
576 {
577  bool frac(false);
578  double sign(1.), value(0.), scale(1.);
579 
580  while (isspace(*p))
581  ++p;
582 
583  // Get sign, if any.
584  if (*p == '-')
585  {
586  sign = -1.;
587  ++p;
588  }
589  else if (*p == '+')
590  ++p;
591 
592  // Get digits before decimal point or exponent, if any.
593  while (isdigit(*p))
594  {
595  value = value * 10. + (*p - '0');
596  ++p;
597  }
598 
599  // Get digits after decimal point, if any.
600  if (*p == '.')
601  {
602  ++p;
603  double pow10(10.);
604  while (isdigit(*p))
605  {
606  value += (*p - '0') / pow10;
607  pow10 *= 10.;
608  ++p;
609  }
610  }
611 
612  // Handle exponent, if any.
613  if (toupper(*p) == 'E')
614  {
615  unsigned int expon(0);
616  ++p;
617 
618  // Get sign of exponent, if any.
619  if (*p == '-')
620  {
621  frac = true;
622  ++p;
623  }
624  else if (*p == '+')
625  ++p;
626 
627  // Get digits of exponent, if any.
628  while (isdigit(*p))
629  {
630  expon = expon * 10 + (*p - '0');
631  ++p;
632  }
633  if (expon > 308)
634  expon = 308;
635 
636  // Calculate scaling factor.
637  while (expon >= 50)
638  {
639  scale *= 1E50;
640  expon -= 50;
641  }
642  while (expon >= 8)
643  {
644  scale *= 1E8;
645  expon -= 8;
646  }
647  while (expon > 0)
648  {
649  scale *= 10.0;
650  expon -= 1;
651  }
652  }
653 
654  // Return signed and scaled floating point result.
655  return sign * (frac ? (value / scale) : (value * scale));
656 }
