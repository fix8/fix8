697 {

715  return tdays * 86400 + (ltm.tm_hour + utcdiff) * 3600 + ltm.tm_min * 60 + ltm.tm_sec;

731 inline size_t date_time_format(const Tickval& tickval, char *to, const TimeIndicator ind)
732 {
733  tm result;
734  tickval.as_tm(result);
735  const char *start(to);
736 
737  if (ind > _time_with_ms)
738  {
739  format0(result.tm_year + 1900, to, 4);
740  to += 4;
741  format0(result.tm_mon + 1, to, 2);
742  to += 2;
743  if (ind == _short_date_only)
744  return to - start;
745  format0(result.tm_mday, to, 2);
746  to += 2;
747  if (ind == _date_only)
748  return to - start;
749  *to++ = '-';
750  }
751 
752  format0(result.tm_hour, to, 2);
753  to += 2;
754  *to++ = ':';
755  format0(result.tm_min, to, 2);
756  to += 2;
757  *to++ = ':';
758  format0(result.tm_sec, to, 2);
759  to += 2;
760 
761  if (ind == _time_with_ms || ind == _with_ms)
762  {
763  *to++ = '.';
764  format0(tickval.msecs(), to, 3);
765  to += 3;
766  }
767 
768  return to - start;
769 }

775 inline Tickval::ticks date_time_parse(const char *ptr, size_t len)
776 {
777  Tickval::ticks result(Tickval::noticks);
778  int millisecond(0);
779  tm tms = {};
780 
781  ptr += parseDate(ptr, 4, tms.tm_year);
782  tms.tm_year -= 1900;
783  ptr += parseDate(ptr, 2, tms.tm_mon);
784  --tms.tm_mon;
785  ptr += parseDate(ptr, 2, tms.tm_mday);
786  ptr += parseDate(++ptr, 2, tms.tm_hour);
787  ptr += parseDate(++ptr, 2, tms.tm_min);
788  ptr += parseDate(++ptr, 2, tms.tm_sec);
789  switch(len)
790  {
791  case 21: //_with_ms: // 19981231-23:59:59.123
792  parseDate(++ptr, 3, millisecond);
793  result = millisecond * Tickval::million; // drop through
794  case 17: //: // 19981231-23:59:59
795  result += time_to_epoch(tms) * Tickval::billion;
796  break;
797  default:
798  break;
799  }
800 
801  return result;
802 }
