75  {

96  {

717 template<const unsigned short field>
718 inline size_t Field<UTCTimestamp, field>::DateTimeFormat(const Poco::DateTime& dateTime, char *to, const MillisecondIndicator ind) const
719 {
720  format0(dateTime.year(), to, 4);
721  format0(dateTime.month(), to + 4, 2);
722  format0(dateTime.day(), to + 6, 2);
723  to[8] = '-';
724 
725  format0(dateTime.hour(), to + 9, 2);
726  to[11] = ':';
727 
728  format0(dateTime.minute(), to + 12, 2);
729  to[14] = ':';
730 
731  format0(dateTime.second(), to + 15, 2);
732 
733  if(ind != _with_ms)
734  return _sec_only; //length of "YYYYMMDD-HH:MM:SS"
735 
736  to[17] = '.';
737  format0(dateTime.millisecond(), to + 18, 3);
738  return _with_ms; //length of "YYYYMMDD-HH:MM:SS.MMM"
739 }

741 template<const unsigned short field>
742 inline void Field<UTCTimestamp, field>::DateTimeParse(const std::string& from, Poco::DateTime& dateTime, const MillisecondIndicator ind) const
743 {
744  std::string::const_iterator it(from.begin());
745  short year(0), month(0), day(0), hour(0), minute(0), second(0), millisecond(0);
746 
747  parseDate(it, 4, year);
748  parseDate(it, 2, month);
749  parseDate(it, 2, day);
750  parseDate(++it, 2, hour);
751  parseDate(++it, 2, minute);
752  parseDate(++it, 2, second);
753 
754  if(ind == _with_ms)
755  parseDate(++it, 3, millisecond);
756 
757  if (Poco::DateTime::isValid(year, month, day, hour, minute, second, millisecond))
758  dateTime.assign(year, month, day, hour, minute, second, millisecond);
759 }
