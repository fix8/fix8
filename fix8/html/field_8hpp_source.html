666 inline time_t time_to_epoch (const tm& ltm, int utcdiff=0)
667 {
668  static const int mon_days[] = {0,
669  31,
670  31 + 28,
671  31 + 28 + 31,
672  31 + 28 + 31 + 30,
673  31 + 28 + 31 + 30 + 31,
674  31 + 28 + 31 + 30 + 31 + 30,
675  31 + 28 + 31 + 30 + 31 + 30 + 31,
676  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31,
677  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30,
678  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31,
679  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31 + 30,
680  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31 + 30 + 31
681  };
682 
683  const int tyears(ltm.tm_year ? ltm.tm_year - 70 : 0); // tm->tm_year is from 1900.
684  const int tdays(mon_days[ltm.tm_mon] + (ltm.tm_mday ? ltm.tm_mday - 1 : 0) + tyears * 365 + (tyears + 2) / 4);
685  return tdays * 86400 + (ltm.tm_hour + utcdiff) * 3600 + ltm.tm_min * 60 + ltm.tm_sec;
686 }

701 inline size_t date_time_format(const Tickval& tickval, char *to, const TimeIndicator ind)
702 {
703  tm result;
704  tickval.as_tm(result);
705  const char *start(to);
706 
707  if (ind > _time_with_ms)
708  {
709  format0(result.tm_year + 1900, to, 4);
710  to += 4;
711  format0(result.tm_mon + 1, to, 2);
712  to += 2;
713  if (ind == _short_date_only)
714  return to - start;
715  format0(result.tm_mday, to, 2);
716  to += 2;
717  if (ind == _date_only)
718  return to - start;
719  *to++ = '-';
720  }
721 
722  format0(result.tm_hour, to, 2);
723  to += 2;
724  *to++ = ':';
725  format0(result.tm_min, to, 2);
726  to += 2;
727  *to++ = ':';
728  format0(result.tm_sec, to, 2);
729  to += 2;
730 
731  if (ind == _time_with_ms || ind == _with_ms)
732  {
733  *to++ = '.';
734  format0(tickval.msecs(), to, 3);
735  to += 3;
736  }
737 
738  return to - start;
739 }

745 inline Tickval::ticks date_time_parse(const char *ptr, size_t len)
746 {
747  Tickval::ticks result(Tickval::noticks);
748  int millisecond(0);
749  tm tms = {};
750 
751  ptr += parseDate(ptr, 4, tms.tm_year);
752  tms.tm_year -= 1900;
753  ptr += parseDate(ptr, 2, tms.tm_mon);
754  --tms.tm_mon;
755  ptr += parseDate(ptr, 2, tms.tm_mday);
756  ++ptr;
757  ptr += parseDate(ptr, 2, tms.tm_hour);
758  ++ptr;
759  ptr += parseDate(ptr, 2, tms.tm_min);
760  ++ptr;
761  ptr += parseDate(ptr, 2, tms.tm_sec);
762  switch(len)
763  {
764  case 21: //_with_ms: // 19981231-23:59:59.123
765  parseDate(++ptr, 3, millisecond);
766  result = millisecond * Tickval::million; // drop through
767  case 17: //: // 19981231-23:59:59
768  result += time_to_epoch(tms) * Tickval::billion;
769  break;
770  default:
771  break;
772  }
773 
774  return result;
775 }
