00057 struct RealmBase
00058 {
00059    enum RealmType { dt_range, dt_set };
00060 
00061    const void *_range;
00062    RealmType _dtype;
00063    FieldTrait::FieldType _ftype;

00065    const char * const *_descriptions;
00066 
00071    template<typename T>
00072    bool is_valid(const T& what) const
00073    {
00074       const T *rng(static_cast<const T*>(_range));
00075       return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);
00076    }
00077 
00082    template<typename T>
00083    int get_rlm_idx(const T& what) const
00084    {
00085       if (_dtype == dt_set)
00086       {
00087          const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));
00088          return res != rng + _sz ? res - rng : -1;
00089       }
00090       return 0;
00091    }
