00096    if (dplaces)

00101       oss << setw(3 + dplaces) << setfill('0') << setprecision(dplaces) << secs;
00102    }
00103    else
00104       oss << setfill('0') << setw(2) << tim.tm_sec;
00105    return result = oss.str();
