343 {
344  unsigned num(preload_count);
345  if (!num)
346  {
347  cout << "Enter number of NewOrderSingle msgs to batch preload:";

352  }
353  while (num > 0)
354  {
355  unsigned cnt(0);
356  for (; cnt < num && cnt < batch_size; ++cnt)
357  {
358  static unsigned oid(10000);
359  ostringstream oistr;
360  oistr << "ord" << ++oid << '-' << num;
361 
362  TEX::NewOrderSingle *ptr(new TEX::NewOrderSingle);
363  TEX::Price *prc(new TEX::Price(1. + RandDev::getrandom(500.)));
364  prc->set_precision(3);
365 
366  *ptr << new TEX::Symbol("BHP")
367  << new TEX::HandlInst(TEX::HandlInst_AUTOMATED_EXECUTION_ORDER_PRIVATE_NO_BROKER_INTERVENTION)
368  << new TEX::OrdType(TEX::OrdType_LIMIT)
369  << new TEX::Side(TEX::Side_BUY)
370  << new TEX::TimeInForce(TEX::TimeInForce_FILL_OR_KILL)
371  << new TEX::TransactTime
372  << new TEX::ClOrdID(oistr.str())
373  << prc
374  << new TEX::OrderQty(1 + RandDev::getrandom(10000));
375 
376  _session.push(ptr);
377  }
378  cout << _session.cached() << " NewOrderSingle msgs preloaded." << endl;
379  num -= cnt;
380  send_all_preloaded();
381  }
382 
383  return true;

533 bool tex_router_server::operator() (const TEX::NewOrderSingle *msg) const
534 {
535  static unsigned oid(0), eoid(0);
536  TEX::OrderQty qty;
537  TEX::Price price;
538  msg->get(qty);
539  msg->get(price);
540 
541  TEX::ExecutionReport *er(new TEX::ExecutionReport);
542  msg->copy_legal(er);
543 
544  ostringstream oistr;
545  oistr << "ord" << ++oid;
546  *er << new TEX::OrderID(oistr.str())
547  << new TEX::ExecType(TEX::ExecType_NEW);
548  unsigned ordResult(RandDev::getrandom(3));
549  switch (ordResult)
550  {
551  default:
552  case 0:
553  *er << new TEX::OrdStatus(TEX::OrdStatus_NEW);
554  break;
555  case 1:
556  *er << new TEX::OrdStatus(TEX::OrdStatus_CANCELED);
557  break;
558  case 2:
559  *er << new TEX::OrdStatus(TEX::OrdStatus_REJECTED);
560  break;
561  }
562 
563  *er << new TEX::LeavesQty(qty())
564  << new TEX::CumQty(0)
565  << new TEX::AvgPx(0)
566  << new TEX::LastCapacity('5')
567  << new TEX::ReportToExch('Y')
568  << new TEX::ExecTransType(TEX::ExecTransType_NEW)
569  << new TEX::ExecID(oistr.str());
570 
571  _session.send(er);
572 
573  if (ordResult == 0)
574  {
575  unsigned remaining_qty(qty()), cum_qty(0);
576  while (remaining_qty > 0)
577  {
578  unsigned trdqty(1 + RandDev::getrandom(remaining_qty));
579  remaining_qty -= trdqty;
580  cum_qty += trdqty;
581  TEX::ExecutionReport *ner(new TEX::ExecutionReport);
582  msg->copy_legal(ner);
583  ostringstream eistr;
584  eistr << "exec" << ++eoid;
585 
586  *ner << new TEX::ExecID(eistr.str())
587  << new TEX::OrderID(oistr.str())
588  << new TEX::ExecType(TEX::ExecType_NEW)
589  << new TEX::OrdStatus(remaining_qty == trdqty ? TEX::OrdStatus_FILLED : TEX::OrdStatus_PARTIALLY_FILLED)
590  << new TEX::LeavesQty(remaining_qty)
591  << new TEX::CumQty(cum_qty)
592  << new TEX::ExecTransType(TEX::ExecTransType_NEW)
593  << new TEX::AvgPx(price());
594 
595  _session.send(ner);
596  }
597  }
598 
599  return true;
600 }
