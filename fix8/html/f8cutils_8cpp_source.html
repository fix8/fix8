94 ostream *open_ofile(const string& odir, const string& fname, string& target)
95 {
96  if (!exist(odir))
97  return 0;
98  ostringstream ofs;
99  string odirect(odir);
100  ofs << CheckAddTrailingSlash(odirect) << fname;
101  target = ofs.str();
102  scoped_ptr<ofstream> os(new ofstream(target.c_str()));
103  if (!*os)
104  {
105  cerr << "Error opening file \'" << target << '\'';
106  if (errno)
107  cerr << " (" << strerror(errno) << ')';
108  cerr << endl;
109 
110  return 0;
111  }
112 
113  return os.release();
114 }

132 int load_fix_version (XmlElement& xf, Ctxt& ctxt)
133 {
134  const XmlElement *fix(xf.find("fix"));
135  if (!fix)
136  {
137  cerr << "No fix header element found in " << shortName << endl;
138  return -1;
139  }
140 
141  string major, minor, revision("0"), type("FIX");
142 
143  if (!fix->GetAttr("major", major) || !fix->GetAttr("minor", minor))
144  {
145  cerr << "Missing required attributes (major/minor) from fix header in " << shortName << endl;
146  return -1;
147  }
148 
149  if (!fix->GetAttr("revision", revision))
150  fix->GetAttr("servicepack", revision);
151  fix->GetAttr("type", type);
152 
153  // fix version: <Major:1><Minor:1><Revision:2> eg. 4.2r10 is 4210
154  ctxt._version = get_value<int>(major) * 1000 + get_value<int>(minor) * 100 + get_value<int>(revision);
155  if (type == "FIX" && ctxt._version < 4000)
156  {
157  cerr << "Unsupported FIX version " << ctxt._version << " from fix header in " << shortName << endl;
158  return -1;
159  }
160 
161  ostringstream ostr;
162  ostr << type << ctxt._version;
163  ctxt._systemns = ostr.str();
164  if (ctxt._fixns.empty())
165  ctxt._fixns = ctxt._systemns;
166  ctxt._clname = prefix;
167 
168  ostr.str("");
169  ostr << type << '.' << major << '.' << minor;
170  ctxt._beginstr = ostr.str();
171 
172  return 0;
173 }

193 void process_value_enums(FieldSpecMap::const_iterator itr, ostream& ost_hpp, ostream& ost_cpp)
194 {
195  string typestr("const ");
196  if (FieldTrait::is_int(itr->second._ftype))
197  typestr += "int ";
198  else if (FieldTrait::is_char(itr->second._ftype))
199  typestr += "char ";
200  else if (FieldTrait::is_float(itr->second._ftype))
201  typestr += "double ";
202  else if (FieldTrait::is_string(itr->second._ftype))
203  typestr += "f8String ";
204  else
205  return;
206 
207  ost_cpp << typestr << itr->second._name << "_realm[] = " << endl << spacer << "{ ";
208  unsigned cnt(0);
209  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
210  {
211  if (cnt)
212  ost_cpp << ", ";
213  ost_cpp << *ditr->first;
214  string transdesc(ditr->second);
215  // replace any illegal c++ identifier characters
216  InPlaceReplaceInSet(ident_set, transdesc, '_');
217  ost_hpp << typestr << itr->second._name << '_';
218  if (transdesc.empty())
219  ost_hpp << *ditr->first;
220  else
221  ost_hpp << transdesc;
222  if (ditr->first->is_range())
223  {
224  if (cnt == 0)
225  ost_hpp << "_lower";
226  else if (cnt == 1)
227  ost_hpp << "_upper";
228  }
229  ost_hpp << '(' << *ditr->first << ");" << endl;
230  ++cnt;
231  }
232  ost_hpp << "const size_t " << itr->second._name << "_realm_els(" << itr->second._dvals->size() << ");" << endl;
233  ost_cpp << " };" << endl;
234 
235  ost_cpp << "const char *" << itr->second._name << "_descriptions[] = " << endl << spacer << "{ ";
236  cnt = 0;
237  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
238  {
239  if (cnt)
240  ost_cpp << ", ";
241  ost_cpp << '"' << ditr->second << '"';
242  ++cnt;
243  }
244  ost_cpp << " };" << endl;
245 }

248 int process_message_fields(const std::string& where, const XmlElement *xt, FieldTraits& fts, const FieldToNumMap& ftonSpec,
249  FieldSpecMap& fspec, const Components& compon)
250 {
251  unsigned processed(0);
252  XmlElement::XmlSet flist;
253  if (xt->find(where, flist))
254  {
255  for(XmlElement::XmlSet::const_iterator fitr(flist.begin()); fitr != flist.end(); ++fitr)
256  {
257  string fname, required;
258  if ((*fitr)->GetAttr("name", fname) && (*fitr)->GetAttr("required", required))
259  {
260  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(fname));
261  FieldSpecMap::iterator fs_itr;
262  if (ftonItr == ftonSpec.end() || (fs_itr = fspec.find(ftonItr->second)) == fspec.end())
263  {
264  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
265  ++glob_errors;
266  continue;
267  }
268 
269  string compname;
270  unsigned compidx((*fitr)->GetAttr("component", compname) ? lookup_component(compon, compname) : 0);
271 
272  // add FieldTrait
273  if (!fts.add(FieldTrait(fs_itr->first, fs_itr->second._ftype, (*fitr)->GetSubIdx(), required == "Y", false, compidx)))
274  {
275  if (!nowarn)
276  cerr << shortName << ':' << recover_line(**fitr) << ": warning: Could not add trait object '" << fname << "' (duplicate ?)" << endl;
277  ++glob_warnings;
278  }
279  else
280  {
281  process_special_traits(fs_itr->first, fts);
282  ++processed;
283  fs_itr->second.set_used();
284  }
285  }
286  else
287  {
288  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
289  ++glob_errors;
290  }
291  }
292  }
293 
294  return processed;
295 }
