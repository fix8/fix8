131 int load_fix_version (XmlElement& xf, Ctxt& ctxt)
132 {
133  const XmlElement *fix(xf.find("fix"));
134  if (!fix)
135  {
136  cerr << "No fix header element found in " << shortName << endl;
137  return -1;
138  }
139 
140  string major, minor, revision("0"), type("FIX");
141 
142  if (!fix->GetAttr("major", major) || !fix->GetAttr("minor", minor))
143  {
144  cerr << "Missing required attributes (major/minor) from fix header in " << shortName << endl;
145  return -1;
146  }
147 
148  if (!fix->GetAttr("revision", revision))
149  fix->GetAttr("servicepack", revision);
150  fix->GetAttr("type", type);
151 
152  // fix version: <Major:1><Minor:1><Revision:2> eg. 4.2r10 is 4210
153  ctxt._version = GetValue<int>(major) * 1000 + GetValue<int>(minor) * 100 + GetValue<int>(revision);
154  if (type == "FIX" && ctxt._version < 4000)
155  {
156  cerr << "Unsupported FIX version " << ctxt._version << " from fix header in " << shortName << endl;
157  return -1;
158  }
159 
160  ostringstream ostr;
161  ostr << type << ctxt._version;
162  ctxt._systemns = ostr.str();
163  if (ctxt._fixns.empty())
164  ctxt._fixns = ctxt._systemns;
165  ctxt._clname = prefix;
166 
167  ostr.str("");
168  ostr << type << '.' << major << '.' << minor;
169  ctxt._beginstr = ostr.str();
170 
171  return 0;
172 }

191 void process_value_enums(FieldSpecMap::const_iterator itr, ostream& ost_hpp, ostream& ost_cpp)
192 {
193  string typestr("const ");
194  if (FieldTrait::is_int(itr->second._ftype))
195  typestr += "int ";
196  else if (FieldTrait::is_char(itr->second._ftype))
197  typestr += "char ";
198  else if (FieldTrait::is_float(itr->second._ftype))
199  typestr += "double ";
200  else if (FieldTrait::is_string(itr->second._ftype))
201  typestr += "f8String ";
202  else
203  return;
204 
205  ost_cpp << typestr << itr->second._name << "_realm[] = " << endl << spacer << "{ ";
206  unsigned cnt(0);
207  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
208  {
209  if (cnt)
210  ost_cpp << ", ";
211  ost_cpp << *ditr->first;
212  string transdesc(ditr->second);
213  // replace any illegal c++ identifier characters
214  InPlaceReplaceInSet(ident_set, transdesc, '_');
215  ost_hpp << typestr << itr->second._name << '_';
216  if (transdesc.empty())
217  ost_hpp << *ditr->first;
218  else
219  ost_hpp << transdesc;
220  if (ditr->first->is_range())
221  {
222  if (cnt == 0)
223  ost_hpp << "_lower";
224  else if (cnt == 1)
225  ost_hpp << "_upper";
226  }
227  ost_hpp << '(' << *ditr->first << ");" << endl;
228  ++cnt;
229  }
230  ost_hpp << "const size_t " << itr->second._name << "_realm_els(" << itr->second._dvals->size() << ");" << endl;
231  ost_cpp << " };" << endl;
232 
233  ost_cpp << "const char *" << itr->second._name << "_descriptions[] = " << endl << spacer << "{ ";
234  cnt = 0;
235  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
236  {
237  if (cnt)
238  ost_cpp << ", ";
239  ost_cpp << '"' << ditr->second << '"';
240  ++cnt;
241  }
242  ost_cpp << " };" << endl;
243 }

246 int process_message_fields(const std::string& where, const XmlElement *xt, FieldTraits& fts, const FieldToNumMap& ftonSpec,
247  FieldSpecMap& fspec, const unsigned subpos)
248 {
249  unsigned processed(0);
250  XmlElement::XmlSet flist;
251  if (xt->find(where, flist))
252  {
253  for(XmlElement::XmlSet::const_iterator fitr(flist.begin()); fitr != flist.end(); ++fitr)
254  {
255  string fname, required;
256  if ((*fitr)->GetAttr("name", fname) && (*fitr)->GetAttr("required", required))
257  {
258  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(fname));
259  FieldSpecMap::iterator fs_itr;
260  if (ftonItr == ftonSpec.end() || (fs_itr = fspec.find(ftonItr->second)) == fspec.end())
261  {
262  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
263  ++glob_errors;
264  continue;
265  }
266 
267  // add FieldTrait
268  if (!fts.add(FieldTrait(fs_itr->first, fs_itr->second._ftype, (*fitr)->GetSubIdx(), required == "Y", false, subpos)))
269  cerr << shortName << ':' << recover_line(**fitr) << ": error: Could not add trait object " << fname << endl;
270  else
271  {
272  process_special_traits(fs_itr->first, fts);
273  ++processed;
274  fs_itr->second.set_used();
275  }
276  }
277  else
278  {
279  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
280  ++glob_errors;
281  }
282  }
283  }
284 
285  return processed;
286 }

295 void process_ordering(MessageSpecMap& mspec)
296 {
297  for (MessageSpecMap::const_iterator mitr(mspec.begin()); mitr != mspec.end(); ++mitr)
298  {
299  FieldTraitOrder mo;
300  for (Presence::const_iterator flitr(mitr->second._fields.get_presence().begin());
301  flitr != mitr->second._fields.get_presence().end(); ++flitr)
302  mo.insert(FieldTraitOrder::value_type(&*flitr));
303 
304  unsigned cnt(0);
305  for (FieldTraitOrder::iterator fto(mo.begin()); fto != mo.end(); ++fto)
306  (*fto)->_pos = ++cnt;
307 
308  process_group_ordering(mitr->second);
309  }
310 }

327  process_group_ordering(gitr->second);

360 RealmObject *RealmObject::create(const string& from, FieldTrait::FieldType ftype, bool isRange)
361 {
362  if (FieldTrait::is_int(ftype))
363  return new TypedRealm<int>(GetValue<int>(from), isRange);
364  if (FieldTrait::is_char(ftype))
365  return new CharRealm(from[0], isRange);
366  if (FieldTrait::is_float(ftype))
367  return new TypedRealm<double>(GetValue<double>(from), isRange);
368  if (FieldTrait::is_string(ftype))
369  return new StringRealm(from, isRange);
370  return 0;
