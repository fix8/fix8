72 unsigned MessageBase::extract_header(const f8String& from, char *len, char *mtype)
73 {
74  const char *dptr(from.data());
75  const size_t flen(from.size());
76  char tag[MAX_MSGTYPE_FIELD_LEN], val[MAX_FLD_LENGTH];
77  unsigned s_offset(0), result;
78 
79  if ((result = extract_element(dptr, flen, tag, val)))
80  {
81  if (*tag != '8')
82  return 0;
83  s_offset += result;
84  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, len)))
85  {
86  if (*tag != '9')
87  return 0;
88  s_offset += result;
89  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, mtype)))
90  {
91  if (*tag != '3' || *(tag + 1) != '5')
92  return 0;
93  s_offset += result;
94  }
95  }
96  }
97  return s_offset;
98 }

108 unsigned MessageBase::decode(const f8String& from, const unsigned offset)
109 {
110  unsigned s_offset(offset), result;
111  const unsigned fsize(from.size());
112  const char *dptr(from.data());
113  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
114 
115  for (unsigned pos(_pos.size()); s_offset <= fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
116  {
117  const unsigned tv(fast_atoi<unsigned>(tag));
118  const BaseEntry *be(_ctx.find_be(tv));
119  if (!be)
120  throw InvalidField(tv);
121  Presence::const_iterator itr(_fp.get_presence().end());
122  if (!_fp.has(tv, itr))
123  break;
124  s_offset += result;
125  if (_fp.get(tv, itr, FieldTrait::present))
126  {
127  if (!_fp.get(tv, itr, FieldTrait::automatic))
128  throw DuplicateField(tv);
129  }
130  else
131  {
132  add_field(tv, itr, ++pos, be->_create(val, be->_rlm, -1), false);
133  if (_fp.is_group(tv, itr))
134  s_offset = decode_group(tv, from, s_offset);
135  }
136  }
137 
138  const unsigned short missing(_fp.find_missing());
139  if (missing)
140  {
141  const BaseEntry *tbe(_ctx.find_be(missing));
142  ostringstream ostr;
143  ostr << tbe->_name << " (" << missing << ')';
144  throw MissingMandatoryField(ostr.str());
145  }
146 
147  return s_offset;
148 }

151 unsigned MessageBase::decode_group(const unsigned short fnum, const f8String& from, const unsigned offset)
152 {
153  unsigned s_offset(offset), result;
154  GroupBase *grpbase(find_group(fnum));
155  if (!grpbase)
156  throw InvalidRepeatingGroup(fnum);
157  const unsigned fsize(from.size());
158  const char *dptr(from.data());
159  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
160 
161  for (bool ok(true); ok && s_offset < fsize; )
162  {
163  scoped_ptr<MessageBase> grp(grpbase->create_group());
164 
165  for (unsigned pos(0); s_offset < fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
166  {
167  const unsigned tv(fast_atoi<unsigned>(tag));
168  Presence::const_iterator itr(grp->_fp.get_presence().end());
169  if (grp->_fp.get(tv, itr, FieldTrait::present)) // already present; next group?
170  break;
171  if (pos == 0 && grp->_fp.getPos(tv, itr) != 1) // first field in group is mandatory
172  throw MissingRepeatingGroupField(tv);
173  const BaseEntry *be(_ctx.find_be(tv));
174  if (!be)
175  throw InvalidField(tv);
176  if (!grp->_fp.has(tv, itr)) // field not found in sub-group - end of repeats?
177  {
178  ok = false;
179  break;
180  }
181  s_offset += result;
182  grp->add_field(tv, itr, ++pos, be->_create(val, be->_rlm, -1), false);
183  grp->_fp.set(tv, itr, FieldTrait::present); // is present
184  if (grp->_fp.is_group(tv, itr)) // nested group
185  s_offset = grp->decode_group(tv, from, s_offset);
186  }
187 
188  const unsigned short missing(grp->_fp.find_missing());
189  if (missing)
190  {
191  const BaseEntry *tbe(_ctx.find_be(missing));
192  ostringstream ostr;
193  ostr << tbe->_name << " (" << missing << ')';
194  throw MissingMandatoryField(ostr.str());
195  }
196  *grpbase << grp.release();
197  }
198 
199  return s_offset;
200 }

209 Message *Message::factory(const F8MetaCntx& ctx, const f8String& from)
210 {
211  Message *msg(0);
212  char mtype[MAX_MSGTYPE_FIELD_LEN] = {}, len[MAX_MSGTYPE_FIELD_LEN] = {};
213  if (extract_header(from, len, mtype))
214  {
215  const unsigned mlen(fast_atoi<unsigned>(len));
216  const BaseMsgEntry *bme(ctx._bme.find_ptr(mtype));
217  if (!bme)
218  throw InvalidMessage(mtype);
219  msg = bme->_create();
220 #if defined CODECTIMING
221  IntervalTimer itm;
222 #endif
223  msg->decode(from);
224 #if defined CODECTIMING
225  _decode_timings._cpu_used += itm.Calculate().AsDouble();
226  ++_decode_timings._msg_count;
227 #endif
228 
229  msg->_header->get_body_length()->set(mlen);
230  msg->_header->get_msg_type()->set(mtype);
231 #if defined POPULATE_METADATA
232  msg->check_set_rlm(fitr->second);
233 #endif
234 
235  const char *pp(from.data() + from.size() - 7);
236  if (*pp != '1' || *(pp + 1) != '0') // 10=XXX^A
237  throw InvalidMessage(from);
238  if (!ctx.has_flag(F8MetaCntx::noverifychksum)) // permit chksum calculation to be skipped
239  {
240  const f8String chksum(pp + 3, 3);
241  msg->_trailer->get_check_sum()->set(chksum);
242  const unsigned chkval(fast_atoi<unsigned>(chksum.c_str())), mchkval(calc_chksum(from, 0, from.size() - 7));
243  if (chkval != mchkval)
244  throw BadCheckSum(mchkval);
245  }
246  }
247  else
248  {
249  //cerr << "Message::factory throwing" << endl;
250  throw InvalidMessage(from);
251  }
252 
253  return msg;
254 }

257 // copy all fields from this message to 'to' where the field is legal for 'to' and it is not
258 // already present in 'to'; includes repeating groups;
259 // if force, copy all fields regardless, replacing any existing, adding any new
260 unsigned MessageBase::copy_legal(MessageBase *to, bool force) const
261 {
262  unsigned copied(0);
263  for (Presence::const_iterator itr(_fp.get_presence().begin()); itr != _fp.get_presence().end(); ++itr)
264  {
265  if (itr->_field_traits & FieldTrait::present && (force || (to->_fp.has(itr->_fnum) && !to->_fp.get(itr->_fnum))))
266  {
267  if (itr->_field_traits & FieldTrait::group)
268  {
269  GroupBase *gb(find_group(itr->_fnum)), *gb1(to->find_group(itr->_fnum));
270 
271  for (GroupElement::const_iterator gitr(gb->_msgs.begin()); gitr != gb->_msgs.end(); ++gitr)
272  {
273  MessageBase *grc(gb1->create_group());
274  (*gitr)->copy_legal(grc, force);
275  *gb1 += grc;
276  }
277  }
278 
279  BaseField *nf(get_field(itr->_fnum)->copy());
280 #if defined POPULATE_METADATA
281  to->check_set_rlm(nf);
282 #endif
283  Presence::const_iterator fpitr(_fp.get_presence().end());
284  if (force && to->_fp.get(itr->_fnum, fpitr, FieldTrait::present))
285  delete to->replace(itr->_fnum, fpitr, nf);
286  else
287  to->add_field(nf);
288  ++copied;
289  }
290  }
291 
292  return copied;
293 }

296 size_t MessageBase::encode(char *to) const
297 {
298  const char *where(to);
299  for (Positions::const_iterator itr(_pos.begin()); itr != _pos.end(); ++itr)
300  {
301 #if defined POPULATE_METADATA
302  check_set_rlm(itr->second);
303 #endif
304  Presence::const_iterator fpitr(_fp.get_presence().end());
305  if (!_fp.get(itr->second->_fnum, fpitr, FieldTrait::suppress)) // some fields are not encoded until unsuppressed (eg. checksum)
306  {
307  to += itr->second->encode(to);
308  if (_fp.get(itr->second->_fnum, fpitr, FieldTrait::group))
309  to += encode_group(itr->second->_fnum, to);
310  }
311  }
312 
313  return to - where;
314 }

363 size_t Message::encode(char **hmsg_store) const
364 {
365  char *moffs(*hmsg_store + HEADER_CALC_OFFSET), *msg(moffs);
366 
367 #if defined CODECTIMING
368  IntervalTimer itm;
369 #endif
370 
371  if (!_header)
372  throw MissingMessageComponent("header");
373  _header->get_msg_type()->set(_msgType);
374  msg += _header->encode(msg); // start
375  msg += MessageBase::encode(msg);
376  if (!_trailer)
377  throw MissingMessageComponent("trailer");
378  msg += _trailer->encode(msg);
379  const size_t msgLen(msg - moffs); // checksummable msglength
380  const size_t hlen(_ctx._preamble_sz + (msgLen < 10 ? 1 : msgLen < 100 ? 2 : msgLen < 1000 ? 3 : 4));
381  char *hmsg(moffs - hlen);
382  *hmsg_store = hmsg;
383 
384  if (!_header->get_begin_string())
385  throw MissingMandatoryField(Common_BeginString);
386  _header->_fp.clear(Common_BeginString, FieldTrait::suppress);
387  hmsg += _header->get_begin_string()->encode(hmsg);
388 #if defined MSGRECYCLING
389  _header->_fp.set(Common_BeginString, FieldTrait::suppress); // in case we want to reuse
390 #endif
391 
392  if (!_header->get_body_length())
393  throw MissingMandatoryField(Common_BodyLength);
394  _header->_fp.clear(Common_BodyLength, FieldTrait::suppress);
395  _header->get_body_length()->set(msgLen);
396  hmsg += _header->get_body_length()->encode(hmsg);
397 #if defined MSGRECYCLING
398  _header->_fp.set(Common_BodyLength, FieldTrait::suppress); // in case we want to reuse
399 #endif
400 
401  if (!_trailer->get_check_sum())
402  throw MissingMandatoryField(Common_CheckSum);
403  _trailer->get_check_sum()->set(fmt_chksum(calc_chksum(moffs - hlen, msgLen + hlen)));
404  _trailer->_fp.clear(Common_CheckSum, FieldTrait::suppress);
405  msg += _trailer->get_check_sum()->encode(msg);
406 #if defined MSGRECYCLING
407  _trailer->_fp.set(Common_CheckSum, FieldTrait::suppress); // in case we want to reuse
408 #endif
409 
410 #if defined CODECTIMING
411  _encode_timings._cpu_used += itm.Calculate().AsDouble();
412  ++_encode_timings._msg_count;
413 #endif
414 
415  *msg = 0;
416  return msg - *hmsg_store;
417 }
