51 unsigned MessageBase::extract_header(const f8String& from, char *len, char *mtype)
52 {
53  const char *dptr(from.data());
54  const unsigned flen(static_cast<unsigned>(from.size()));
55  char tag[MAX_MSGTYPE_FIELD_LEN], val[MAX_FLD_LENGTH];
56  unsigned s_offset(0), result;
57 
58  if ((result = extract_element(dptr, flen, tag, val)))
59  {
60  if (*tag != '8')
61  return 0;
62  s_offset += result;
63  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, len)))
64  {
65  if (*tag != '9')
66  return 0;
67  s_offset += result;
68  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, mtype)))
69  {
70  if (*tag != '3' || *(tag + 1) != '5')
71  return 0;
72  s_offset += result;
73  }
74  }
75  }
76  return s_offset;
77 }

87 unsigned MessageBase::decode(const f8String& from, unsigned s_offset, unsigned ignore, bool permissive_mode)
88 {
89  const unsigned fsize(static_cast<unsigned>(from.size()) - ignore), npos(0xffffffff);
90  unsigned pos(static_cast<unsigned>(_pos.size())), last_valid_pos(npos);
91  const char *dptr(from.data());
92  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
93  size_t last_valid_offset(0);
94 
95  for (unsigned result; s_offset <= fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
96  {
97  const unsigned short tv(fast_atoi<unsigned short>(tag));
98  Presence::const_iterator itr(_fp.get_presence().find(tv));
99  if (itr == _fp.get_presence().end())
100  {
101  if (permissive_mode)
102  {
103  if (last_valid_pos == npos)
104  {
105  last_valid_pos = pos;
106  last_valid_offset = s_offset;
107  }
108  _unknown.append(dptr + s_offset, result);
109  s_offset += result;
110  continue;
111  }
112  break;
113  }
114  s_offset += result;
115  if (itr->_field_traits.has(FieldTrait::present))
116  {
117  if (!itr->_field_traits.has(FieldTrait::automatic))
118  throw DuplicateField(tv);
119  }
120  else
121  {
122  const BaseEntry *be(_ctx.find_be(tv));
123  if (!be)
124  throw UnknownField(tv);
125  BaseField *bf(be->_create._do(val, be->_rlm, -1));
126  add_field_decoder(tv, ++pos, bf);
127  itr->_field_traits.set(FieldTrait::present);
128  // check if repeating group and num elements > 0
129  if (itr->_field_traits.has(FieldTrait::group) && has_group_count(bf))
130  s_offset = decode_group(nullptr, tv, from, s_offset, ignore);
131  }
132  }
133 
134  const unsigned short missing(_fp.find_missing());
135  if (missing)
136  {
137  const BaseEntry *tbe(_ctx.find_be(missing));
138  ostringstream ostr;
139  ostr << tbe->_name << " (" << missing << ')';
140  throw MissingMandatoryField(ostr.str());
141  }
142 
143  return permissive_mode && last_valid_pos == pos ? static_cast<unsigned>(last_valid_offset) : s_offset;
144 }

147 unsigned MessageBase::decode_group(GroupBase *grpbase, const unsigned short fnum, const f8String& from,
148  unsigned s_offset, unsigned ignore)
149 {
150  unsigned result;
151  if (!(grpbase = find_add_group(fnum, grpbase)))
152  throw InvalidRepeatingGroup(fnum, FILE_LINE);
153  const unsigned fsize(static_cast<unsigned>(from.size()) - ignore);
154  const char *dptr(from.data());
155  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
156 
157  for (bool ok(true); ok && s_offset < fsize; )
158  {
159  unique_ptr<MessageBase> grp(grpbase->create_group(false)); // shallow create
160 
161  for (unsigned pos(0); s_offset < fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
162  {
163  const unsigned tv(fast_atoi<unsigned>(tag));
164  Presence::const_iterator itr(grp->_fp.get_presence().end());
165  if (grp->_fp.get(tv, itr, FieldTrait::present)) // already present; next group?
166  break;
167  if (pos == 0 && grp->_fp.getPos(tv, itr) != 1) // first field in group is mandatory
168  throw MissingRepeatingGroupField(tv);
169  const BaseEntry *be(_ctx.find_be(tv));
170  if (!be || !grp->_fp.has(tv, itr)) // unknown field or field not found in sub-group - end of repeats?
171  {
172  ok = false;
173  break;
174  }
175  s_offset += result;
176  BaseField *bf(be->_create._do(val, be->_rlm, -1));
177  grp->add_field(tv, itr, ++pos, bf, false);
178  grp->_fp.set(tv, itr, FieldTrait::present); // is present
179  // nested group (check if not zero elements)
180  if (grp->_fp.is_group(tv, itr) && has_group_count(bf))
181  s_offset = grp->decode_group(grpbase, tv, from, s_offset, ignore);
182  }
183 
184  const unsigned short missing(grp->_fp.find_missing());
185  if (missing)
186  {
187  const BaseEntry *tbe(_ctx.find_be(missing));
188  ostringstream ostr;
189  ostr << tbe->_name << " (" << missing << ')';
190  throw MissingMandatoryField(ostr.str());
191  }
192  *grpbase << grp.release();
193  }
194 
195  return s_offset;
196 }

205 Message *Message::factory(const F8MetaCntx& ctx, const f8String& from, bool no_chksum, bool permissive_mode)
206 {
207  char mtype[MAX_MSGTYPE_FIELD_LEN] {}, len[MAX_MSGTYPE_FIELD_LEN] {};
208  const unsigned hlen(extract_header(from, len, mtype));
209 
210  if (!hlen)
211  {
212  glout_debug << "Message::factory throwing";
213  throw InvalidMessage(from, FILE_LINE);
214  }
215 
216  const unsigned mlen(fast_atoi<unsigned>(len));
217  const BaseMsgEntry *bme(ctx._bme.find_ptr(mtype));
218  if (!bme)
219  throw InvalidMessage(mtype, FILE_LINE);
220  Message *msg(bme->_create._do(false)); // shallow create
221 #if defined CODECTIMING
222  IntervalTimer itm;
223 #endif
224  msg->decode(from, hlen, 7, permissive_mode); // skip already decoded mandatory 8, 9, 35 and 10
225 #if defined CODECTIMING
226  _decode_timings._cpu_used += itm.Calculate().AsDouble();
227  ++_decode_timings._msg_count;
228 #endif
229 
230  msg->_header->get_body_length()->set(mlen);
231  msg->_header->get_msg_type()->set(mtype);
232 #if defined POPULATE_METADATA
233  msg->check_set_rlm(fitr->second);
234 #endif
235 
236  const char *pp(from.data() + from.size() - 7); // FIXME: assumes supplied string is one complete message only
237  if (*pp != '1' || *(pp + 1) != '0') // 10=XXX^A
238  throw InvalidMessage(from, FILE_LINE);
239  if (!no_chksum) // permit chksum calculation to be skipped
240  {
241  const f8String chksum(pp + 3, 3);
242  msg->_trailer->get_check_sum()->set(chksum);
243  const unsigned chkval(fast_atoi<unsigned>(chksum.c_str())), mchkval(calc_chksum(from, 0, static_cast<unsigned>(from.size()) - 7));
244  if (chkval != mchkval)
245  throw BadCheckSum(mchkval);
246  }
247 
248  return msg;
249 }

252 // copy all fields from this message to 'to' where the field is legal for 'to' and it is not
253 // already present in 'to'; includes repeating groups;
254 // a deep constructed target message is required
255 // if force, copy all fields regardless, replacing any existing, adding any new
256 unsigned MessageBase::copy_legal(MessageBase *to, bool force) const
257 {
258  unsigned copied{};
259  for (const auto& pp : _fp.get_presence())
260  {
261  if (pp._field_traits & FieldTrait::present && (force || (to->_fp.has(pp._fnum) && !to->_fp.get(pp._fnum))))
262  {
263  GroupBase *gb;
264  if (pp._field_traits & FieldTrait::group && (gb = find_group(pp._fnum)))
265  {
266  GroupBase *gb1(to->find_group(pp._fnum));
267  for (const auto *qq : gb->_msgs)
268  {
269  MessageBase *grc(gb1->create_group(true));
270  copied += qq->copy_legal(grc, force);
271  *gb1 += grc;
272  }
273  }
274 
275  BaseField *nf(get_field(pp._fnum)->copy());
276 #if defined POPULATE_METADATA
277  to->check_set_rlm(nf);
278 #endif
279  Presence::const_iterator fpitr(_fp.get_presence().end());
280  if (force && to->_fp.get(pp._fnum, fpitr, FieldTrait::present))
281  delete to->replace(pp._fnum, fpitr, nf);
282  else
283  to->add_field(nf);
284  ++copied;
285  }
286  }
287 
288  return copied;
289 }

292 // move all fields from this message to 'to' where the field is legal for 'to' and it is not
293 // already present in 'to'; includes repeating groups;
294 // if force, move all fields regardless, replacing any existing
295 // not thread safe
296 unsigned MessageBase::move_legal(MessageBase *to, bool force)
297 {
298  unsigned moved{};
299  for (const auto& pp : _fp.get_presence())
300  {
301  if (pp._field_traits & FieldTrait::present && (force || (to->_fp.has(pp._fnum) && !to->_fp.get(pp._fnum))))
302  {
303  if (pp._field_traits & FieldTrait::group)
304  {
305  auto gitr(_groups.find(pp._fnum));
306  GroupBase *gb1(to->find_group(pp._fnum));
307  if (gb1)
308  delete to->replace(pp._fnum, gitr->second);
309  else
310  *to += gitr->second;
311  gitr->second = nullptr;
312  }
313 
314  auto itr(_fields.find(pp._fnum));
315  Presence::const_iterator fpitr(_fp.get_presence().end());
316  if (force && to->_fp.get(pp._fnum, fpitr, FieldTrait::present))
317  delete to->replace(pp._fnum, fpitr, itr->second);
318  else
319  to->add_field(itr->second);
320  itr->second = nullptr;
321  ++moved;
322  }
323  }
324  clear_positions();
325 
326  return moved;
327 }

330 size_t MessageBase::encode(char *to) const
331 {
332  const char *where(to);
333  for (const auto& pp : _pos)
334  {
335 #if defined POPULATE_METADATA
336  check_set_rlm(pp.second);
337 #endif
338  Presence::const_iterator fpitr(_fp.get_presence().end());
339  if (!_fp.get(pp.second->_fnum, fpitr, FieldTrait::suppress)) // some fields are not encoded until unsuppressed (eg. checksum)
340  {
341  to += pp.second->encode(to);
342  if (fpitr->_field_traits.has(FieldTrait::group) && has_group_count(pp.second))
343  to += encode_group(pp.second->_fnum, to);
344  }
345  }
346 
347  if (_unknown.size())
348  to += _unknown.copy(to, _unknown.size());
349 
350  return to - where;
351 }

403 size_t Message::encode(char **hmsg_store) const
404 {
405  char *moffs(*hmsg_store + HEADER_CALC_OFFSET), *msg(moffs);
406 
407 #if defined CODECTIMING
408  IntervalTimer itm;
409 #endif
410 
411  if (!_header)
412  throw MissingMessageComponent("header");
413  _header->get_msg_type()->set(_msgType);
414 
415 #if defined RAW_MSG_SUPPORT
416  msg += (_begin_payload = _header->encode(msg)); // start
417 #if defined PREENCODE_MSG_SUPPORT
418  if (_preencode_len)
419  {
420  ::memcpy(msg, _preencode.data(), _payload_len =_preencode_len);
421  msg += _preencode_len;
422  }
423  else
424 #endif
425  msg += (_payload_len = MessageBase::encode(msg));
426 #else
427  msg += _header->encode(msg); // start
428 #if defined PREENCODE_MSG_SUPPORT
429  if (_preencode_len)
430  {
431  ::memcpy(msg, _preencode.data(), _preencode_len);
432  msg += _preencode_len;
433  }
434  else
435 #endif
436  msg += MessageBase::encode(msg);
437 #endif
438 
439  if (!_trailer)
440  throw MissingMessageComponent("trailer");
441  msg += _trailer->encode(msg);
442  const size_t msgLen(msg - moffs); // checksummable msglength
443  const size_t hlen(_ctx._preamble_sz +
444  (msgLen < 10 ? 1 : msgLen < 100 ? 2 : msgLen < 1000 ? 3 : msgLen < 10000 ? 4 :
445  msgLen < 100000 ? 5 : msgLen < 1000000 ? 6 : 7));
446  char *hmsg(moffs - hlen);
447  *hmsg_store = hmsg;
448 
449  if (!_header->get_begin_string())
450  throw MissingMandatoryField(Common_BeginString);
451  _header->_fp.clear(Common_BeginString, FieldTrait::suppress);
452  hmsg += _header->get_begin_string()->encode(hmsg);
453 
454  if (!_header->get_body_length())
455  throw MissingMandatoryField(Common_BodyLength);
456  _header->_fp.clear(Common_BodyLength, FieldTrait::suppress);
457 
458  _header->get_body_length()->set(static_cast<int>(msgLen));
459  hmsg += _header->get_body_length()->encode(hmsg);
460 
461  if (!_trailer->get_check_sum())
462  throw MissingMandatoryField(Common_CheckSum);
463  _trailer->get_check_sum()->set(fmt_chksum(calc_chksum(moffs - hlen, msgLen + hlen)));
464  _trailer->_fp.clear(Common_CheckSum, FieldTrait::suppress);
465  msg += _trailer->get_check_sum()->encode(msg);
466 
467 #if defined CODECTIMING
468  _encode_timings._cpu_used += itm.Calculate().AsDouble();
469  ++_encode_timings._msg_count;
470 #endif
471 
472  *msg = 0;
473  const size_t rlen(msg - *hmsg_store);
474 #if defined RAW_MSG_SUPPORT
475  _rawmsg.assign(*hmsg_store, rlen);
476 #endif
477  return rlen;
478 }
