129  int sockRead(char *where, size_t sz)
130  {
131  if (static_cast<size_t>(_read_buffer_wptr - _read_buffer_rptr) < sz)
132  realSockRead(sz, _max_msg_len);
133  sz = std::min((size_t)(_read_buffer_wptr-_read_buffer_rptr), sz);
134  memcpy(where, _read_buffer_rptr, sz);
135  _read_buffer_rptr += sz;
136  const size_t shift(_max_msg_len);
137  if (static_cast<size_t>(_read_buffer_rptr - _read_buffer) >= shift)
138  {
139  memcpy(_read_buffer, &_read_buffer[shift], sizeof(_read_buffer) - shift);
140  _read_buffer_rptr -= shift;
141  _read_buffer_wptr -= shift;
142  }
143  return sz;
144  }
145 
151  int realSockRead(size_t sz, size_t maxsz)
152  {
153  const size_t max_sz(_read_buffer + sizeof(_read_buffer) - _read_buffer_wptr);
154  int maxremaining(std::min(maxsz, max_sz)), remaining(std::min(sz, max_sz));
155  char *ptr(_read_buffer_wptr), *eptr(_read_buffer + sizeof(_read_buffer));
156 
157  int rdsz(0);
158  while (remaining > 0 && _read_buffer_wptr < eptr)
159  {
160  rdsz = _sock->receiveBytes(_read_buffer_wptr, maxremaining);
161  if (rdsz <= 0)
162  {
163  if (errno == EAGAIN
164 #if defined EWOULDBLOCK && EAGAIN != EWOULDBLOCK
165  || errno == EWOULDBLOCK
166 #endif
167  )
168  continue;
169  throw PeerResetConnection("sockRead: connection gone");
170  }
171  _read_buffer_wptr += rdsz;
172  remaining -= rdsz;
173  maxremaining -= rdsz;
174  }
175  return _read_buffer_wptr - ptr;
176  }

288  size_t write_batch(const std::vector<Message *>& msgs, bool destroy)
289  {
290  if (msgs.empty())
291  return 0;
292  if (msgs.size() == 1)
293  return write(msgs.front(), destroy) ? 1 : 0;
294  size_t result(0);
295  f8_scoped_spin_lock guard(_con_spl);
296  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()), litr(eitr-1); itr != eitr; ++itr)
297  {
298  Message* msg = *itr;
299  msg->set_end_of_batch(itr == litr);
300  if (_pmodel == pm_pipeline) // pipeline mode ignores destroy flag
301  {
302  _msg_queue.try_push(msg);
303  ++result;
304  continue;
305  }
306  if (_session.send_process(msg))
307  ++result;
308  }
309  if (destroy && _pmodel != pm_pipeline)
310  {
311  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()); itr != eitr; ++itr)
312  {
313  scoped_ptr<Message> smsg(*itr);
314  }
315  }
317  return result;
318  }

347  int send(const char *data, size_t remaining, bool nb=false)
348  {
349  unsigned wrdone(0);
350 
351  while (remaining > 0)
352  {
353  const int wrtSz(_sock->sendBytes(data + wrdone, remaining));
354  if (wrtSz < 0)
355  {
356  if (errno == EAGAIN
357 #if defined EWOULDBLOCK && EAGAIN != EWOULDBLOCK
358  || errno == EWOULDBLOCK
359 #endif
360  )
361  {
362  if (nb)
363  return wrdone;
364  continue;
365  }
366  throw PeerResetConnection("send: connection gone");
367  }
368

370  remaining -= wrtSz;
371  }
