348  size_t write_batch(const std::vector<Message *>& msgs, bool destroy)
349  {
350  if (msgs.empty())
351  return 0;
352  if (msgs.size() == 1)
353  return write(msgs.front(), destroy) ? 1 : 0;
354  size_t result(0);
355  f8_scoped_spin_lock guard(_con_spl);
356  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()), litr(eitr-1); itr != eitr; ++itr)
357  {
358  Message* msg = *itr;
359  msg->set_end_of_batch(itr == litr);
360  if (_pmodel == pm_pipeline) // pipeline mode ignores destroy flag
361  {
362  _msg_queue.try_push(msg);
363  ++result;
364  continue;
365  }
366  if (_session.send_process(msg))
367  ++result;
368  }
369  if (destroy && _pmodel != pm_pipeline)
370  {
371  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()); itr != eitr; ++itr)
372  {
373  scoped_ptr<Message> smsg(*itr);
374  }
375  }
377  return result;
378  }
