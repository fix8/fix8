143  virtual inline bool pop(void ** ptr) {
144  if (skipdnode || P) return ff_node::pop(ptr);
145 
146  // gets the peers involved in one single communication
147  const int sendingPeers=com.getToWait();
148 
149  svector<msg_t*>* v[sendingPeers];
150  for(int i=0;i<sendingPeers;++i) {
151  msg_t hdr;
152  int sender=-1;
153  if (!com.gethdr(hdr, sender)) {
154  error("dnode:pop: ERROR: receiving header from peer");
155  return false;
156  }
157  if (isEos(static_cast<char *>(hdr.getData()))) {
158  //std::cerr << "DNODE gets DEOS\n";
159  if (++neos==com.getDescriptor()->getPeers()) {
160  com.done();
161  *ptr = (void*)FF_EOS;
162  neos=0;
163  return true;
164  }
165  if (sendingPeers==1) i=-1; // reset for index
166  continue;
167  }
168  uint32_t len = *static_cast<uint32_t*>(hdr.getData());
169  register int ventry = (sendingPeers==1)?0:sender;
170  prepare(v[ventry], len, sender);
171  assert(v[ventry]->size() == len);
172 
173  for(size_t j=0;j<len;++j)
174  if (!com.get(*(v[ventry]->operator[](j)),sender)) {
175  error("dnode:pop: ERROR: receiving data from peer");
176  return false;
177  }
178  }
179  com.done();
180 
181  unmarshalling(v, sendingPeers, *ptr);
182  return true;
183  }
