00050 gzstreambuf* gzstreambuf::open(const char* name, int open_mode)
00051 {
00052     if (is_open())
00053         return 0;
00054     mode = open_mode;
00055     // no input append nor read/write mode
00056     if ((mode & std::ios::ate) || ((mode & std::ios::in) && (mode & std::ios::out)))
00057         return 0;
00058     char fmode[10], *fmodeptr = fmode;
00059     if (mode & std::ios::in)
00060         *fmodeptr++ = 'r';
00061     else if (mode & std::ios::app)
00062         *fmodeptr++ = 'a';
00063     else if (mode & std::ios::out)
00064         *fmodeptr++ = 'w';
00065     *fmodeptr++ = 'b';

00068         return 0;

00111 int gzstreambuf::flush_buffer()
