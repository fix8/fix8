380 int load_fields(XmlElement& xf, FieldSpecMap& fspec)
381 {
382  int fieldsLoaded(0);
383 
384  XmlElement::XmlSet flist;
385  if (!xf.find("fix/fields/field", flist))
386  {
387  cerr << "error: No fields found in " << shortName << endl;
388  return 0;
389  }
390 
391  for(XmlElement::XmlSet::const_iterator itr(flist.begin()); itr != flist.end(); ++itr)
392  {
393  string number, name, type;
394  if ((*itr)->GetAttr("number", number) && (*itr)->GetAttr("name", name) && (*itr)->GetAttr("type", type))
395  {
396  trim(number);
397  trim(name);
398  trim(type);
399  InPlaceStrToUpper(type);
400  FieldTrait::FieldType ft(FieldSpec::_baseTypeMap.find_value(type));
401  pair<FieldSpecMap::iterator, bool> result;
402  if (ft != FieldTrait::ft_untyped)
403  result = fspec.insert(FieldSpecMap::value_type(get_value<unsigned>(number), FieldSpec(name, ft)));
404  else
405  {
406  cerr << shortName << ':' << recover_line(**itr) << ": warning: Unknown field type: " << type << " in " << name << endl;
407  ++glob_warnings;
408  continue;
409  }
410 
411  (*itr)->GetAttr("description", result.first->second._description);
412  (*itr)->GetAttr("comment", result.first->second._comment);
413 
414  ++fieldsLoaded;
415 
416  XmlElement::XmlSet realmlist;
417  if ((*itr)->find("field/value", realmlist))
418  {
419  for(XmlElement::XmlSet::const_iterator ditr(realmlist.begin()); ditr != realmlist.end(); ++ditr)
420  {
421  string enum_str, description;
422  if ((*ditr)->GetAttr("enum", enum_str))
423  {
424  if (!(*ditr)->GetAttr("description", description) || description.empty())
425  description = enum_str; // use enum if no description supplied
426 
427  if (!result.first->second._dvals)
428  result.first->second._dvals = new RealmMap;
429  string lower, upper;
430  bool isRange((*ditr)->GetAttr("range", lower) && (lower == "lower" || upper == "upper"));
431  RealmObject *realmval(RealmObject::create(enum_str, ft, isRange));
432  if (isRange)
433  result.first->second._dtype = RealmBase::dt_range;
434  if (realmval)
435  result.first->second._dvals->insert(RealmMap::value_type(realmval, description));
436  }
437  else
438  {
439  cerr << shortName << ':' << recover_line(**itr) << ": error: Value element missing required attributes." << endl;
440  ++glob_errors;
441  }
442  }
443  }
444  }
445  else
446  {
447  cerr << shortName << ':' << recover_line(**itr) << ": error: Field element missing required attributes" << endl;
448  ++glob_errors;
449  }
450  }
451 
452  return fieldsLoaded;
453 }

475

477  {

483  // lookup msgtype realm - all messages must have corresponding entry here
484  FieldSpecMap::const_iterator fsitr(fspec.find(35)); // always 35
485  if (fsitr == fspec.end() || !fsitr->second._dvals)

491  }
492

498  else if ((*itr)->GetTag() == "trailer")

501  {

503  RealmMap::const_iterator ditr(fsitr->second._dvals->find(&srealm));
504  if (ditr == fsitr->second._dvals->end())
505  {
506  cerr << shortName << ':' << recover_line(**itr) << ": error: Message "
507  << name << " does not have corrresponding entry in MsgType field realm. See FAQ for more details." << endl;
508  ++glob_errors;
509  continue;
510  }
511 
512  elname = "message";
513  }
514  else
515  {

517  ++glob_errors;

522  mspec.insert(MessageSpecMap::value_type(msgtype, MessageSpec(name, msgcat % "admin"))));

534 
535  (*itr)->GetAttr("comment", result.first->second._comment);

549 unsigned parse_groups(MessageSpec& ritr, XmlElement::XmlSet::const_iterator& itr, const string& name,
550  const FieldToNumMap& ftonSpec, FieldSpecMap& fspec, XmlElement::XmlSet& grplist, const Components& compon)
551 {

555  {
556  string gname, required;
557  if ((*gitr)->GetAttr("name", gname) && (*gitr)->GetAttr("required", required))
558  {
559  string compname;
560  unsigned compidx((*gitr)->GetAttr("component", compname) ? lookup_component(compon, compname) : 0);
561 
562  // add group FieldTrait
563  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(gname));
564  FieldSpecMap::const_iterator fs_itr;
565  if (ftonItr != ftonSpec.end() && (fs_itr = fspec.find(ftonItr->second)) != fspec.end())
566  {

568  required == "Y", true, compidx)))
569  {
570  cerr << "error: Could not add group trait object " << gname << endl;
571  ++glob_errors;
572  }
573  else
574  {
575  fs_itr->second._used = true; // we always assume group count fields are used
576  pair<GroupMap::iterator, bool> gresult(
577  ritr._groups.insert(GroupMap::value_type(fs_itr->first, MessageSpec(gname))));
578  process_message_fields("group/field", *gitr, gresult.first->second._fields, ftonSpec, fspec, compon);
579  XmlElement::XmlSet comlist;
580  ++result;
581  if ((*gitr)->find("group/group", comlist))
582  result += parse_groups(gresult.first->second, gitr, gname, ftonSpec, fspec, comlist, compon);
583  }
584  }
585  else
586  {
587  cerr << shortName << ':' << recover_line(**itr)
588  << ": error: Could not locate group Field " << gname << " from known field types in " << shortName << endl;
589  ++glob_errors;
590  continue;
591  }
592  }
593  else
594  {
595  cerr << shortName << ':' << recover_line(**itr) << ": error: Group element missing required attributes" << endl;
596  ++glob_errors;
597  }
598  }
599 
600  return result;
601 }
