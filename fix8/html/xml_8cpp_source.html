00474    {
00475       string whatv, replv;

00478          replv = "&";

00483       else if (whatv == "apos")

00485       else if (whatv == "quot")

00488          break;   // cannot be reached
00489 
00490       rCX_.Replace(match, what, replv);
00491    }
00492 
00493    while (rCE_.SearchString(match, what, 2) == 2)  // translate Numeric character references &#x12d; or &#12;
00494    {
00495       string whatv;

00497       istringstream istr(whatv);
00498       int value;
00499       if (whatv[0] == 'x')
00500       {
00501          istr.ignore();

00503       }
00504       else
00505          istr >> dec >> value;
