170 #ifdef __APPLE__
171 // A simple spinlock that spins up to 100K times and then does a sched_yield to back-off.
172 // unlock() could just set _lock to false BUT that assumes that the spinlock was locked
173 // to begin with, which may not be the case. Therefore this implementation has the
174 // advantage of causing your thread to spin if you try to unlock something that is
175 // not locked, which I would think is a logic error that the caller should fix.
176 // The choice of 100K was arbitrary. The right way to set that parameter would be
177 // to keep track of how big x gets before thread starvation occurs and use that number.
178 // That number is going to be target- and use-case-dependent though.
179 class f8_spin_lock
180 {
181  bool _isLocked;
182 public:
183  f8_spin_lock() : _isLocked(false) {}
184  ~f8_spin_lock() {}
185 
186  void lock()
187  {
188  register int x = 0;
189  while(!__sync_bool_compare_and_swap(&_isLocked, false, true))
190  {
191  if(++x >= 100000)
192  {
193  x = 0;
194  sched_yield();
195  }
196  }
197  }
198  bool try_lock() { return _isLocked; }
199  void unlock()
200  {
201  register int x = 0;
202  while(!__sync_bool_compare_and_swap(&_isLocked, true, false))
203  {
204  if(++x >= 100000)
205  {
206  x = 0;
207  sched_yield();
208  }
209  }
210  }
211 };
