183  const unsigned recv_seqnum, const f8String davi)
184 {
185  if (_logger)
186  _logger->purge_thread_codes();
187 
188  if (_plogger)
189  _plogger->purge_thread_codes();
190 
191  _control.clear(shutdown);
192  log("Starting session");
193  _connection = connection; // takes owership
194  if (!_connection->connect()) // if already connected returns true
195  return -1;
196  if (_connection->get_role() == Connection::cn_acceptor)
197  atomic_init(States::st_wait_for_logon); // important for server that this is done before connect
198  _connection->start();
199  log("Session connected");
200 
201  if (_connection->get_role() == Connection::cn_initiator)
202  {
203  atomic_init(States::st_not_logged_in);
204  if (_loginParameters._reset_sequence_numbers)
205  _next_send_seq = _next_receive_seq = 1;
206  else
207  {
208  recover_seqnums();
209  if (send_seqnum)
210  _next_send_seq = send_seqnum;
211  if (recv_seqnum)
212  _next_receive_seq = recv_seqnum;
213  }
214 
215  send(generate_logon(_connection->get_hb_interval(), davi));
216  do_state_change(States::st_logon_sent);
217  }
218  else
219  {
220  if (send_seqnum)
221  _req_next_send_seq = send_seqnum;
222  if (recv_seqnum)
223  _req_next_receive_seq = recv_seqnum;
224  }
225 
226  if (_sf && (_schedule = _sf->create_session_schedule(_sf->_ses)))
227  {
228  if (_connection->get_role() == Connection::cn_initiator)
229  {
230  ostringstream ostr;
231  ostr << *_schedule;
232  log(ostr.str());
233  }
234  _timer.schedule(_session_scheduler, 1000); // check every second
235  }
236 
237  if (wait) // wait for
238  {
239  _connection->join();
240  }
241 
242  return 0;
243 }

280 bool Session::process(const f8String& from)
281 {
282  unsigned seqnum(0);
283 
284  try
285  {
286  const f8String::size_type fpos(from.find("34="));
287  if (fpos == f8String::npos)
288  {
289  //cerr << "Session::process throwing for " << from << endl;
290  throw InvalidMessage(from);
291  }
292 
293  seqnum = fast_atoi<unsigned>(from.data() + fpos + 3, default_field_separator);
294 
295  bool retry_plog(false);
296  if (_plogger && _plogger->has_flag(Logger::inbound))
297  {
298  if (_state != States::st_wait_for_logon)
299  plog(from, 1);
300  else
301  retry_plog = true;
302  }
303 
304  const Message *msg(Message::factory(_ctx, from, _loginParameters._no_chksum_flag, _loginParameters._permissive_mode_flag));
305  if (!msg)
306  {
307  GlobalLogger::log("Fatal: factory failed to generate a valid message");
308  return false;
309  }
310 
311  if ((_control & printnohb) && msg->get_msgtype() != Common_MsgType_HEARTBEAT)
312  cout << *msg << endl;
313  else if (_control & print)
314  cout << *msg << endl;
315 
316  bool result(false), admin_result(msg->is_admin() ? handle_admin(seqnum, msg) : true);
317  if (msg->get_msgtype().size() > 1)
318  goto application_call;
319  else switch(msg->get_msgtype()[0])
320  {
321  default:
322 application_call:
323  if (activation_check(seqnum, msg))
324  result = handle_application(seqnum, msg);
325  break;
326  case Common_MsgByte_HEARTBEAT:
327  result = handle_heartbeat(seqnum, msg);
328  break;
329  case Common_MsgByte_TEST_REQUEST:
330  result = handle_test_request(seqnum, msg);
331  break;
332  case Common_MsgByte_RESEND_REQUEST:
333  result = handle_resend_request(seqnum, msg);
334  break;
335  case Common_MsgByte_REJECT:
336  result = handle_reject(seqnum, msg);
337  break;
338  case Common_MsgByte_SEQUENCE_RESET:
339  result = handle_sequence_reset(seqnum, msg);
340  break;
341  case Common_MsgByte_LOGOUT:
342  result = handle_logout(seqnum, msg);
343  break;
344  case Common_MsgByte_LOGON:
345  result = handle_logon(seqnum, msg);
346  break;
347  }
348 
349  ++_next_receive_seq;
350  if (retry_plog)
351  plog(from, 1);
352  if (_persist)
353  {
354  f8_scoped_spin_lock guard(_per_spl, _connection->get_pmodel() == pm_coro);
355  _persist->put(_next_send_seq, _next_receive_seq);
356  //cout << "Persisted:" << _next_send_seq << " and " << _next_receive_seq << endl;
357  }
358  delete msg;
359  return result && admin_result;
360  }
361  catch (f8Exception& e)
362  {
363  //cerr << "process:: f8exception" << ' ' << seqnum << ' ' << e.what() << endl;
364 
365  log(e.what());
366  if (!e.force_logoff())
367  {
368  send(generate_reject(seqnum, e.what()));
369  }
370  else
371  {
372  if (_state == States::st_logon_received && !_loginParameters._silent_disconnect)
373  {
374  send(generate_logout(e.what()), true, 0, true); // so it won't increment
375  do_state_change(States::st_logoff_sent);
376  }
377  stop();
378  }
379  }
380  catch (Poco::Net::NetException& e)
381  {
382  //cerr << "process:: Poco::Net::NetException" << endl;
383  log(e.what());
384  }
385  catch (exception& e)
386  {
387  //cerr << "process:: std::exception" << endl;
388  log(e.what());
389  }
390 
391  return false;
392 }
