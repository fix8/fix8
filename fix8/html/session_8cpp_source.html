169 int Session::start(Connection *connection, bool wait, const unsigned send_seqnum,
170  const unsigned recv_seqnum, const f8String davi)
171 {
172  if (_logger)
173  _logger->purge_thread_codes();
174  if (_plogger)
175  _plogger->purge_thread_codes();
176  _control.clear(shutdown);
177  log("Starting session");
178  _connection = connection; // takes owership
179  if (!_connection->connect()) // if already connected returns true
180  return -1;;
181  if (_connection->get_role() == Connection::cn_acceptor)
182  atomic_init(States::st_wait_for_logon); // important for server that this is done before connect
183  _connection->start();
184  log("Session connected");
185 
186  if (_connection->get_role() == Connection::cn_initiator)
187  {
188  atomic_init(States::st_not_logged_in);
189  if (_loginParamaters._reset_sequence_numbers)
190  _next_send_seq = _next_receive_seq = 1;
191  else
192  {
193  recover_seqnums();
194  if (send_seqnum)
195  _next_send_seq = send_seqnum;
196  if (recv_seqnum)
197  _next_receive_seq = recv_seqnum;
198  }
199 
200  Message *msg(generate_logon(_connection->get_hb_interval(), davi));
201  send(msg);
202  _state = States::st_logon_sent;
203  }
204  else
205  {
206  if (send_seqnum)
207  _req_next_send_seq = send_seqnum;
208  if (recv_seqnum)
209  _req_next_receive_seq = recv_seqnum;
210  }
211 
212  if (wait) // wait for
213  {
214  _connection->join();
215  }
216 
217  return 0;
218 }

254 bool Session::process(const f8String& from)
255 {
256  unsigned seqnum(0);
257 
258  try
259  {
260  const f8String::size_type fpos(from.find("34="));
261  if (fpos == f8String::npos)
262  {
263  //cerr << "Session::process throwing for " << from << endl;
264  throw InvalidMessage(from);
265  }
266 
267  seqnum = fast_atoi<unsigned>(from.data() + fpos + 3, default_field_separator);
268 
269  bool retry_plog(false);
270  if (_plogger && _plogger->has_flag(Logger::inbound))
271  {
272  if (_state != States::st_wait_for_logon)
273  plog(from, 1);
274  else
275  retry_plog = true;
276  }
277 
278  scoped_ptr<Message> msg(Message::factory(_ctx, from));
279  if (!msg.get())
280  {
281  GlobalLogger::log("Fatal: factory failed to generate a valid message");
282  return false;
283  }
284 
285  if ((_control & printnohb) && msg->get_msgtype() != Common_MsgType_HEARTBEAT)
286  cout << *msg << endl;
287  else if (_control & print)
288  cout << *msg << endl;
289  bool result((msg->is_admin() ? handle_admin(seqnum, msg.get()) : true)
290  && (this->*_handlers.find_ref(msg->get_msgtype()))(seqnum, msg.get()));
291  ++_next_receive_seq;
292  if (retry_plog)
293  plog(from, 1);
294  if (_persist)
295  {
296  _persist->put(_next_send_seq, _next_receive_seq);
297  //cout << "Persisted:" << _next_send_seq << " and " << _next_receive_seq << endl;
298  }
299  return result;
300 
301  }
302  catch (f8Exception& e)
303  {
304  //cout << "process:: f8exception" << ' ' << seqnum << ' ' << e.what() << endl;
305 
306  log(e.what());
307  if (!e.force_logoff())
308  {
309  Message *msg(generate_reject(seqnum, e.what()));
310  send(msg);
311  }
312  else
313  {
314  if (_state != States::st_logon_received)
315  {
316  send(generate_logout(), 0, true); // so it won't increment
317  _state = States::st_logoff_sent;
318  }
319  stop();
320  }
321  }
322  catch (exception& e) // also catches Poco::Net::NetException
323  {
324  //cout << "process:: std::exception" << endl;
325 
326  log(e.what());
327  }
328 
329  return false;
330 }

346 bool Session::sequence_check(const unsigned seqnum, const Message *msg)
347 {
348  //cout << "seqnum:" << seqnum << " next_target_seq:" << _next_receive_seq
349  //<< " next_sender_seq:" << _next_send_seq << " state:" << _state << " next_receive_seq:" << _next_receive_seq << endl;
350 
351  if (seqnum > _next_receive_seq)
352  {
353  if (_state == States::st_resend_request_sent)
354  log("Resend request already sent");
355  if (_state == States::st_continuous)
356  {
357  Message *rmsg(generate_resend_request(_next_receive_seq));
358  _state = States::st_resend_request_sent;
359  send(rmsg);
360  }
361  else
362  throw InvalidMsgSequence(seqnum, _next_receive_seq);
363  return false;
364  }
365 
366  if (seqnum < _next_receive_seq)
367  {
368  poss_dup_flag pdf(false);
369  msg->Header()->get(pdf);
370  if (!pdf()) // poss dup not set so bad
371  throw MsgSequenceTooLow(seqnum, _next_receive_seq);
372  sending_time st;
373  msg->Header()->get(st);
374  orig_sending_time ost;
375  if (msg->Header()->get(ost) && ost() > st())
376  {
377  ostringstream ostr;
378  ost.print(ostr);
379  throw BadSendingTime(ostr.str());
380  }
381  }
382 
383  return true;
384 }

387 bool Session::handle_logon(const unsigned seqnum, const Message *msg)
388 {
389  _state = States::st_logon_received;
390  const bool reset_given(msg->have(Common_ResetSeqNumFlag) && msg->get<reset_seqnum_flag>()->get());
391 
392  if (_connection->get_role() == Connection::cn_initiator)
393  {
394  enforce(seqnum, msg);
395  heartbeat_interval hbi;
396  msg->get(hbi);
397  _connection->set_hb_interval(hbi());
398  _state = States::st_continuous;
399  ostringstream ostr;
400  ostr << "Client setting heartbeat interval to " << hbi();
401  log(ostr.str());
402  }
403  else // acceptor
404  {
405  default_appl_ver_id davi;
406  msg->get(davi);
407 
408  sender_comp_id sci;
409  msg->Header()->get(sci);
410  target_comp_id tci;
411  msg->Header()->get(tci);
412  SessionID id(_ctx._beginStr, tci(), sci());
413 
414  // important - these objects can't be created until we have a valid SessionID
415  if (!_logger)
416  _logger = _sf->create_logger(_sf->_ses, Configuration::session_log, &id);
417  if (!_plogger)
418  _plogger = _sf->create_logger(_sf->_ses, Configuration::protocol_log, &id);
419  if (!_persist)
420  _persist = _sf->create_persister(_sf->_ses, &id, reset_given);
421 
422  ostringstream ostr;
423  ostr << "Connection from " << _connection->get_peer_socket_address().toString();
424  log(ostr.str());
425 
426  if (reset_given) // ignore version restrictions on this behaviour
427  {
428  log("Resetting sequence numbers");
429  _next_send_seq = _next_receive_seq = 1;
430  }
431  else
432  {
433  recover_seqnums();
434  if (_req_next_send_seq)
435  _next_send_seq = _req_next_send_seq;
436  if (_req_next_receive_seq)
437  _next_receive_seq = _req_next_receive_seq;
438  }
439 
440  if (authenticate(id, msg))
441  {
442  _sid = id;
443  enforce(seqnum, msg);
444  Message *msg(generate_logon(_connection->get_hb_interval(), davi()));
445  send(msg);
446  _state = States::st_continuous;
447  }
448  else
449  {
450  ostringstream ostr;
451  ostr << id << " failed authentication";
452  log(ostr.str());
453  stop();
454  _state = States::st_session_terminated;
455  return false;
456  }
457  }
458 
459  ostringstream ostr;
460  ostr << "Heartbeat interval is " << _connection->get_hb_interval();
461  log(ostr.str());
462 
463  _timer.schedule(_hb_processor, 1000); // check every second
464 
465  return true;
466 }

502 bool Session::handle_resend_request(const unsigned seqnum, const Message *msg)
503 {
504  enforce(seqnum, msg);
505 
506  if (_state != States::st_resend_request_received)
507  {
508  begin_seq_num begin;
509  end_seq_num end;
510 
511  if (!_persist || !msg->get(begin) || !msg->get(end))
512  send(generate_sequence_reset(_next_send_seq + 1, true));

520  }
521  }
522 
523  return true;

763 bool Session::send_process(Message *msg) // called from the connection (possibly on separate thread)
764 {
765  bool is_dup(msg->Header()->have(Common_PossDupFlag));
766 
767  if (!msg->Header()->have(Common_SenderCompID))
768  *msg->Header() += new sender_comp_id(_sid.get_senderCompID());
769  if (!msg->Header()->have(Common_TargetCompID))
770  *msg->Header() += new target_comp_id(_sid.get_targetCompID());
771 
772  if (msg->Header()->have(Common_MsgSeqNum))
773  {
774  if (is_dup)
775  {
776  if (_loginParamaters._always_seqnum_assign)
777  delete msg->Header()->remove(Common_PossDupFlag);
778  }
779  else
780  {
781  if (!_loginParamaters._always_seqnum_assign)
782  {
783  *msg->Header() += new poss_dup_flag(true);
784  is_dup = true;
785  }
786  }
787 
788  sending_time sendtime;
789  msg->Header()->get(sendtime);
790  *msg->Header() += new orig_sending_time(sendtime());
791 
792  if (_loginParamaters._always_seqnum_assign)
793  //cerr << "send_process: _next_send_seq = " << _next_send_seq << endl;
794  *msg->Header() += new msg_seq_num(msg->get_custom_seqnum() ? msg->get_custom_seqnum() : _next_send_seq);
795  }
796  else
797  {
798  //cerr << "send_process: _next_send_seq = " << _next_send_seq << endl;
799  *msg->Header() += new msg_seq_num(msg->get_custom_seqnum() ? msg->get_custom_seqnum() : _next_send_seq);
800  }
801  *msg->Header() += new sending_time;
802 
803  try
804  {
805  //cout << "Sending:" << *msg;
806  modify_outbound(msg);
807  char output[MAX_MSG_LENGTH + HEADER_CALC_OFFSET], *ptr(output);
808  const size_t enclen(msg->encode(&ptr));
809  if (!_connection->send(ptr, enclen))
810  {
811  ostringstream ostr;
812  ostr << "Message write failed: " << enclen << " bytes";
813  log(ostr.str());
814  return false;
815  }
816  _last_sent.now();
817 
818  if (_plogger && _plogger->has_flag(Logger::outbound))
819  plog(ptr);
820 
821  //cout << "send_process" << endl;
822 
823  if (!is_dup)
824  {
825  if (_persist)
826  {
827  if (!msg->is_admin())
828  _persist->put(_next_send_seq, ptr);
829  _persist->put(_next_send_seq + 1, _next_receive_seq);
830  //cout << "Persisted (send):" << (_next_send_seq + 1) << " and " << _next_receive_seq << endl;
831  }
832 
833  if (!msg->get_custom_seqnum() && !msg->get_no_increment() && msg->get_msgtype() != Common_MsgType_SEQUENCE_RESET)
834  {
835  ++_next_send_seq;
836  //cout << "Seqnum now:" << _next_send_seq << " and " << _next_receive_seq << endl;
837  }
838  }
839  }
840  catch (f8Exception& e)
841  {
842  log(e.what());
843  }
844 
845  return true;
846 }
