101  ctx._load_default_cas, ctx._cipher_list);

103  Poco::Net::SSLManager::instance().initializeClient(phrase_handler, cert_handler, _context);

105  Poco::Net::SSLManager::instance().initializeServer(phrase_handler, cert_handler, _context);

344  {
345  ++_attempts;

347  bool excepted(_failover_cnt == 0);
348  try

350  if (_failover_cnt)

352  std::ostringstream ostr;
353  ostr << "Trying " << _servers[_current]._hostname << '(' << (1 + _servers[_current]._retries) << "), "
354  << _attempts << " attempts so far";

367  this->_sock = new Poco::Net::StreamSocket;
368 #endif

371  this->_session->set_login_parameters(this->_loginParameters);
372  this->_session->set_session_config(this);
373  this->_session->start(this->_cc, true, _send_seqnum, _recv_seqnum, this->_loginParameters._davi());
374  _send_seqnum = _recv_seqnum = 0; // only set seqnums for the first time round
375  }
376  catch (Poco::TimeoutException& e)
377  {
378  this->_session->log(e.what(), Logger::Error);
379  excepted = true;
380  }
381  catch (f8Exception& e)
382  {
383  // std::cerr << e.what() << std::endl;
384  this->_session->log(e.what(), Logger::Error);
385  excepted = true;
386  }
387  catch (Poco::Net::InvalidAddressException& e)

390  excepted = true;
391  }
392 #if POCO_VERSION >= 0x01040000
393  catch (Poco::Net::InvalidSocketException& e)
394  {
395  this->_session->log(e.what(), Logger::Error);
396  excepted = true;
397  }
398 #endif
399  catch (Poco::Net::ServiceNotFoundException& e)
400  {
401  //std::cerr << e.what() << std::endl;
402  this->_session->log(e.what(), Logger::Error);
403  excepted = true;
404  }
405  catch (Poco::Net::ConnectionRefusedException& e)
406  {
407  this->_session->log(e.what(), Logger::Error);
408  excepted = true;
409  }
410  catch (Poco::Net::DNSException& e)
411  {
412  this->_session->log(e.what(), Logger::Error);
413  excepted = true;
414  }
415  catch (Poco::Net::InterfaceNotFoundException& e)
416  {
417  this->_session->log(e.what(), Logger::Error);
418  excepted = true;
419  }
420  catch (Poco::Net::NetException& e) // catch all other NetExceptions
421  {
422  this->_session->log(e.what(), Logger::Error);
423  excepted = true;
424  }
425  catch (std::exception& e)
426  {
427  //std::cout << "process:: std::exception" << endl;
428  this->_session->log(e.what(), Logger::Error);
429  excepted = true;
430  }
431 
432  //std::cout << "operator()():out of try" << std::endl;
433  try
434  {
435  this->_session->stop();
436  }
437  catch (f8Exception& e)
438  {
439  this->_session->log(e.what(), Logger::Error);
440  }
441  catch (Poco::Net::NetException& e)
442  {
443  this->_session->log(e.what(), Logger::Error);
444  }
445  catch (std::exception& e)
446  {
447  this->_session->log(e.what(), Logger::Error);
448  }
449 
450  delete this->_cc;
451  this->_cc = nullptr;
452  delete this->_sock;
453  this->_sock = nullptr;
454 
455  if (!excepted || (_failover_cnt == 0 && _attempts > this->_loginParameters._login_retries))
456  break;
457 
458  if (_failover_cnt)
459  {
460  ++_servers[_current]._retries;
461 
462  while (!_cancellation_token) // FIXME possible endless loop condition
463  {
464  if (_servers[_current]._max_retries && _servers[_current]._retries < _servers[_current]._max_retries)
465  break;
466  _servers[_current]._retries = 0; // reset for next time
467  ++_current %= _failover_cnt;
468  }
469  }
470 
471  hypersleep<h_milliseconds>(this->_loginParameters._login_retry_interval);
472  }
473 
474  _giving_up = true;
475 
476  return 0;
477  }
