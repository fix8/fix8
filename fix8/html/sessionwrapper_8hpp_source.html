139 template<typename T>
140 class ReliableClientSession : public ClientSession<T>
141 {
142  Thread<ReliableClientSession<T> > _thread;
143  unsigned _send_seqnum, _recv_seqnum;
144 
145 public:
147  ReliableClientSession (const F8MetaCntx& ctx, const std::string& conf_file, const std::string& session_name)
148  : ClientSession<T>(ctx, conf_file, session_name, true), _thread(ref(*this)), _send_seqnum(), _recv_seqnum() {}
149 
151  virtual ~ReliableClientSession () {}
152 
158  virtual void start(bool wait, const unsigned send_seqnum=0, const unsigned recv_seqnum=0, const f8String davi=f8String())
159  {
160  _send_seqnum = send_seqnum;
161  _recv_seqnum = recv_seqnum;
162  if (!wait)
163  _thread.Start();
164  else
165  (*this)();
166  }
167 
170  int operator()()
171  {
172  unsigned attempts(0);
173 
174  for (; ++attempts < this->_loginParameters._login_retries; )
175  {
176  try
177  {
178  //std::cout << "operator()():try" << std::endl;
179 
180  this->_sock = new Poco::Net::StreamSocket,
181  this->_cc = new ClientConnection(this->_sock, this->_addr, *this->_session);
182  this->_session->start(this->_cc, true, _send_seqnum, _recv_seqnum, this->_loginParameters._davi());
183  _send_seqnum = _recv_seqnum = 0; // only set seqnums for the first time round
184  }
185  catch(f8Exception& e)
186  {
187  //std::cout << "ReliableClientSession: f8exception" << std::endl;
188  this->_session->log(e.what());
189  }
190  catch (std::exception& e) // also catches Poco::Net::NetException
191  {
192  //cout << "process:: std::exception" << endl;
193  this->_session->log(e.what());
194  }
195 
196  //std::cout << "operator()():out of try" << std::endl;
197  this->_session->stop();
198  delete this->_cc;
199  delete this->_sock;
200  millisleep(this->_loginParameters._login_retry_interval);
201  }
202 
203  return 0;
204  }
205 
207  typedef scoped_ptr<ReliableClientSession<T> > ReliableClient_ptr;
208 };
