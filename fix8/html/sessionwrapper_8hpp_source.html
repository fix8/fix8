157  {
158  if (!wait)
159  _thread.Start();
160  else
161  (*this)();
162  }
163 
166  int operator()()
167  {
168  bool reset_sequence_numbers;
169  unsigned attempts(0), login_retry_interval, login_retries;
170  default_appl_ver_id davi;
171  this->_session->get_login_parameters(login_retry_interval, login_retries, davi, reset_sequence_numbers);
172 
173  for (; ++attempts < login_retries; )
174  {
175  try
176  {
177  //std::cout << "operator()():try" << std::endl;
178 
179  this->_sock = new Poco::Net::StreamSocket,
180  this->_cc = new ClientConnection(this->_sock, this->_addr, *this->_session);
181  this->_session->start(this->_cc, true, 0, 0, davi());
182  }
183  catch(f8Exception& e)
184  {
185  //std::cout << "ReliableClientSession: f8exception" << std::endl;
186  this->_session->log(e.what());
187  }
188 
189  //std::cout << "operator()():out of try" << std::endl;
190  delete this->_cc;
191  delete this->_sock;
192  millisleep(login_retry_interval);
193  }
194 
195  return 0;
196  }
