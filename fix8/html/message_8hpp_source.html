588  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
589  {
590  enum { get_tag, get_value } state(get_tag);
591 
592  for (unsigned ii(0); ii < sz; ++ii)
593  {
594  switch (state)
595  {
596  case get_tag:
597  if (!isdigit(from[ii]))
598  {
599  if (from[ii] != '=')
600  {
601  *val = *tag = 0;
602  return 0;
603  }
604  state = get_value;
605  }
606  else
607  *tag++ = from[ii];
608  break;
609  case get_value:
610  if (from[ii] == default_field_separator)
611  {
612  *val = *tag = 0;
613  return ++ii;
614  }
615  *val++ = from[ii];
616  break;
617  }
618  }
619  *val = *tag = 0;
620  return 0;
621  }
