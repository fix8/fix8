618  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
619  {
620  enum { get_tag, get_value } state(get_tag);
621 
622  for (unsigned ii(0); ii < sz; ++ii)
623  {
624  switch (state)
625  {
626  case get_tag:
627  if (!isdigit(from[ii]))
628  {
629  if (from[ii] != default_assignment_separator)
630  return *val = *tag = 0;
631  state = get_value;
632  }
633  else

635  break;
636  case get_value:
637  if (from[ii] == default_field_separator)

641  }

644  }
645  }
