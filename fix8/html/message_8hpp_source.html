598  static unsigned extract_element(const char *from, const unsigned sz, f8String& tag, f8String& val)
599  {
600  enum { get_tag, get_value } state(get_tag);
601  tag.clear();
602  val.clear();
603 
604  for (unsigned ii(0); ii < sz; ++ii)
605  {
606  switch (state)
607  {
608  case get_tag:
609  if (!isdigit(from[ii]))
610  {
611  if (from[ii] != '=')
612  return 0;
613  state = get_value;
614  }
615  else
616  tag += from[ii];
617  break;
618  case get_value:
619  if (from[ii] == default_field_separator)
620  return ++ii;
621  val += from[ii];
622  break;
623  }
624  }
625  return 0;
626  }
