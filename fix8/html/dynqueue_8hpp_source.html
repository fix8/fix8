284  inline bool cachepush(void * const data) {
285 
286  if (!cache[pwrite]) {
287  /* Write Memory Barrier: ensure all previous memory write
288  * are visible to the other processors before any later
289  * writes are executed. This is an "expensive" memory fence
290  * operation needed in all the architectures with a weak-ordering
291  * memory model where stores can be executed out-or-order
292  * (e.g. Powerpc). This is a no-op on Intel x86/x86-64 CPUs.
293  */
294  WMB();
295  cache[pwrite] = data;
296  pwrite += (pwrite+1 >= cachesize) ? (1-cachesize): 1;
297  return true;
298  }
299  return false;
300  }

305  inline bool cachepop(void ** data) {
306  if (!cache[pread]) return false;
307 
308  *data = cache[pread];
309  cache[pread]=NULL;
310  pread += (pread+1 >= cachesize) ? (1-cachesize): 1;
311  return true;
312  }
