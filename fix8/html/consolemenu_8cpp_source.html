327 void ConsoleMenu::EditMsg(tty_save_state& tty, const FieldTable::Pair *fld, Message *msg) const
328 {
329  string txt;
330  int rval(-1);
331  if (fld->_value._rlm)
332  rval = SelectRealm(fld->_key, fld->_value._rlm);
333  else
334  {
335  _os << endl << fld->_value._name << ": " << flush;
336  GetString(tty, txt);
337  if (msg->get_fp().is_group(fld->_key))
338  {
339  int cnt(GetValue<int>(txt));
340  GroupBase *gb(msg->find_group(fld->_key));
341  if (gb && cnt)
342  {
343  for (int ii(0); ii < cnt; ++ii)
344  {
345  Message *gmsg(static_cast<Message *>(gb->create_group()));
346  const FieldTable::Pair *fld;
347  while((fld = SelectField(gmsg, ii + 1)))
348  EditMsg(tty, fld, gmsg);
349  _os << endl << endl << *static_cast<MessageBase *>(gmsg) << endl;
350  if (get_yn("Add group to msg? (y/n):", true))
351  *gb += gmsg;
352  }
353  }
354  }
355  }
356 
357  BaseField *bf(fld->_value._create(txt, fld->_value._rlm, rval));
358  msg->add_field(bf->get_tag(), msg->get_fp().get_presence().end(), 0, bf, true);
359 }
