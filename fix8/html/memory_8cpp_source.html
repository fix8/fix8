63 MemoryPool::MemoryPool(const size_t numblocks, const size_t blksz) throw(bad_alloc)
64  : _store(), _store_end(), _numblocks(numblocks), _blksz(blksz)
65 {
66  _blknum = 0;
67  long pageSz;
68  if ((pageSz = sysconf(_SC_PAGESIZE)) == -1)
69  pageSz = default_page_sz;
70 
71  if (posix_memalign(&_store, pageSz * sizeof(void *), _blksz * _numblocks))
72  throw bad_alloc();
73  _store_end = static_cast<char*>(_store) + _blksz * _numblocks;
74  _free_list.set_capacity(_numblocks);
75  //cout << "Memorypool allocated " << _blksz * _numblocks << " bytes" << endl;
76 }

79 void *MemoryPool::alloc() throw(bad_alloc)
80 {
81  void *addr;
82  if (!_free_list.try_pop(addr))
83  {
84  addr = static_cast<char *>(_store) + _blksz * _blknum++;
85  if (_blknum >= _numblocks)
86  throw bad_alloc();
87  }
88  return addr;
89 }
