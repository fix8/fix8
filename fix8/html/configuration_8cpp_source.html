66 int Configuration::process()
67 {
68  if (!exist(_xmlfile))
69  throw f8Exception("server config file not found", _xmlfile);
70  if (!_root)
71  throw f8Exception("could not create root xml entity");
72 
73  if (!load_map("fix8/session", _sessions, true))
74  throw f8Exception("could not locate server session in config file", _xmlfile);
75 
76  load_map("fix8/persist", _persisters);
77  load_map("fix8/log", _loggers);
78 
79  return _sessions.size();
80 }

102 Persister *Configuration::create_persister(const XmlElement *from, const SessionID *sid, bool flag) const
103 {
104  string name, type;
105  const XmlElement *which;
106  if (from && from->GetAttr("persist", name) && (which = find_persister(name)) && which->GetAttr("type", type))
107  {
108  if (type == "mem")
109  return new MemoryPersister;
110 
111  string dir("./"), db("persist_db");
112  which->GetAttr("dir", dir);
113  which->GetAttr("db", db);
114 
115  if (sid)
116  db += ('.' + sid->get_senderCompID()() + '.' + sid->get_targetCompID()());
117  else if (which->FindAttr("use_session_id", false))
118  db += ('.' + get_sender_comp_id(from)() + '.' + get_target_comp_id(from)());
119 
120 #if defined HAVE_BDB
121  if (type == "bdb")
122  {
123  scoped_ptr<BDBPersister> result(new BDBPersister);
124  if (result->initialise(dir, db, flag))
125  return result.release();
126  }
127  else
128 #endif
129  if (type == "file")
130  {
131  scoped_ptr<FilePersister> result(new FilePersister(which->FindAttr("rotation", 0)));
132  if (result->initialise(dir, db, flag))
133  return result.release();
134  }
135  }
136 
137  return 0;
138 }

141 Logger *Configuration::create_logger(const XmlElement *from, const Logtype ltype, const SessionID *sid) const
142 {
143  string name;
144  if (from && from->GetAttr(ltype == session_log ? "session_log" : "protocol_log", name))
145  {
146  const XmlElement *which(find_logger(name));
147  if (which)
148  {
149  string type;
150  if (which->GetAttr("type", type)
151  && ((type % "session" && ltype == session_log) || (type % "protocol" && ltype == protocol_log)))
152  {
153  string logname("logname_not_set.log");
154  which->FindAttrRef("filename", logname);
155  trim(logname);
156 
157  if (logname[0] == '|')
158 #ifndef HAVE_POPEN
159  throw f8Exception("popen not supported on your platform");
160 #endif
161  return new PipeLogger(logname, get_logflags(which));
162 
163  RegMatch match;
164  if (_ipexp.SearchString(match, logname, 3) == 3)
165  {
166  f8String ip, port;
167  _ipexp.SubExpr(match, logname, ip, 0, 1);
168  _ipexp.SubExpr(match, logname, port, 0, 2);
169  BCLogger *bcl(new BCLogger(ip, get_value<unsigned>(port), get_logflags(which)));
170  if (*bcl)
171  return bcl;
172  }
173 
174  get_logname(which, logname, sid); // only applies to file loggers
175  return new FileLogger(logname, get_logflags(which), get_logfile_rotation(which));
176  }
177  }
178  }
179 
180  return 0;
181 }

195 Logger::LogFlags Configuration::get_logflags(const XmlElement *from) const
196 {
197  Logger::LogFlags flags;
198  string flags_str;
199  if (from && from->GetAttr("flags", flags_str))
200  {
201  istringstream istr(flags_str);
202  for(char extr[32]; !istr.get(extr, sizeof(extr), '|').eof() || extr[0]; istr.ignore(1))
203  {
204  string result(extr);
205  flags.set(Logger::num_flags, Logger::_bit_names, trim(result));
206  }
207  }
208 
209  return flags;
210 }
