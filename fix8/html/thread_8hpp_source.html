322 #ifdef __APPLE__
323 // A simple spinlock that spins up to 100K times and then does a sched_yield to back-off.
324 // unlock() could just set _lock to false BUT that assumes that the spinlock was locked
325 // to begin with, which may not be the case. Therefore this implementation has the
326 // advantage of causing your thread to spin if you try to unlock something that is
327 // not locked, which I would think is a logic error that the caller should fix.
328 // The choice of 100K was arbitrary. The right way to set that parameter would be
329 // to keep track of how big x gets before thread starvation occurs and use that number.
330 // That number is going to be target- and use-case-dependent though.
331 class f8_spin_lock
332 {
333  bool _isLocked;
334 public:
335  f8_spin_lock() : _isLocked(false) {}
336  ~f8_spin_lock() {}
337 
338  void lock()
339  {
340  register int x = 0;
341  while(!__sync_bool_compare_and_swap(&_isLocked, false, true))
342  {
343  if(++x >= 100000)
344  {
345  x = 0;
346  sched_yield();
347  }
348  }
349  }
350  bool try_lock() { return _isLocked; }
351  void unlock()
352  {
353  register int x = 0;
354  while(!__sync_bool_compare_and_swap(&_isLocked, true, false))
355  {
356  if(++x >= 100000)
357  {
358  x = 0;
359  sched_yield();
360  }
361  }
362  }
363 };
364 #else

433  }

435  bool try_acquire(T& mutex)
436  {
437  bool result(mutex.try_lock());
438  if (result)
439  _local_mutex = &mutex;
440  return result;
441  }
442 
443  void release()
444  {

447  _local_mutex->unlock();
